## scripts/publisher/production_metadata.py contract (PUB-04 D-08)

```python
"""Production metadata — Phase 8 PUB-04 D-08.

4-field TypedDict per Phase 4 ins-platform-policy enforcement contract:
{script_seed, assets_origin, pipeline_version, checksum}.
Injected as HTML comment at end of video description — invisible to viewers,
parseable by Phase 9 analytics via regex r'<!-- production_metadata\\n(\\{.*?\\})\\n-->'.
"""
from __future__ import annotations

import hashlib
import json
from pathlib import Path
from typing import TypedDict

PIPELINE_VERSION = "1.0.0"   # Phase 8 shipping version


class ProductionMetadata(TypedDict):
    script_seed: str
    assets_origin: str
    pipeline_version: str
    checksum: str   # "sha256:<64-hex>"


def compute_checksum(mp4_path: Path) -> str:
    """Stream-hash 64KB chunks → 'sha256:<hexdigest>'."""
    h = hashlib.sha256()
    with Path(mp4_path).open("rb") as f:
        for chunk in iter(lambda: f.read(65536), b""):
            h.update(chunk)
    return f"sha256:{h.hexdigest()}"


def inject_into_description(description: str, meta: ProductionMetadata) -> str:
    """Append '<!-- production_metadata\\n{JSON}\\n-->' — ensure_ascii=False for Korean."""
    required = {"script_seed", "assets_origin", "pipeline_version", "checksum"}
    missing = required - set(meta.keys())
    if missing:
        raise ValueError(f"PUB-04 schema violation: missing fields {missing}")
    block = (
        "\n<!-- production_metadata\n"
        + json.dumps(meta, ensure_ascii=False, separators=(",", ":"))
        + "\n-->"
    )
    return description + block
```

## scripts/publisher/youtube_uploader.py contract (PUB-02 + PUB-05)

```python
"""YouTube Data API v3 uploader — Phase 8 PUB-02 + PUB-05 D-09.

ONLY Data API v3 endpoints used: videos.insert, thumbnails.set, commentThreads.insert.
NO Selenium/webdriver/playwright (AF-8). NO captions.insert (not for end-screens — Pitfall 7).
NO endScreen field (not writable via Data API v3 — YouTube Studio UI only).
defaultAudioLanguage silently dropped (Pitfall 5 — not writable per snippet spec).
"""
from __future__ import annotations

import random
import time
from pathlib import Path
from typing import Any, Protocol

from scripts.publisher.ai_disclosure import build_status_block, assert_synthetic_media_true
from scripts.publisher.exceptions import GitHubRemoteError   # unused — import for namespace discovery
from scripts.publisher.kst_window import assert_in_window
from scripts.publisher.production_metadata import (
    compute_checksum,
    inject_into_description,
    PIPELINE_VERSION,
)
from scripts.publisher.publish_lock import assert_can_publish, record_upload


RETRIABLE_STATUS_CODES = [500, 502, 503, 504]
MAX_RETRIES = 10


class YouTubeClient(Protocol):
    def videos(self) -> Any: ...
    def thumbnails(self) -> Any: ...
    def commentThreads(self) -> Any: ...


def build_insert_body(plan: dict[str, Any]) -> dict[str, Any]:
    """Translate Phase 4 publisher upload plan JSON → Data API v3 body.

    Corrections applied (RESEARCH Pitfalls 5/6/7):
    - containsSyntheticMedia = True (hardcoded via build_status_block) — Pitfall 6
    - defaultAudioLanguage dropped silently — Pitfall 5
    - funnel.end_screen_subscribe_cta is a plan-level intent, NOT an API call — Pitfall 7
    """
    snippet_in = plan.get("snippet", {})
    snippet = {
        "title": snippet_in.get("title", "")[:100],
        "description": snippet_in.get("description", "")[:5000],
        "tags": snippet_in.get("tags", []),
        "categoryId": snippet_in.get("categoryId", "24"),
        "defaultLanguage": snippet_in.get("defaultLanguage", "ko"),
        # NOTE: defaultAudioLanguage is NOT a writable snippet field per official docs
        # — intentionally dropped (Pitfall 5).
    }
    status_in = plan.get("status", {})
    status = build_status_block(
        privacy_status=status_in.get("privacyStatus", "public"),
        embeddable=status_in.get("embeddable", True),
        public_stats_viewable=status_in.get("publicStatsViewable", True),
    )
    # containsSyntheticMedia is hardcoded True inside build_status_block; verify.
    assert_synthetic_media_true(status)
    return {"snippet": snippet, "status": status}


def resumable_upload(insert_request) -> str:
    """Returns videoId on success. Exponential backoff on retriable errors."""
    response = None
    retry = 0
    while response is None:
        try:
            status, response = insert_request.next_chunk()
            if response is not None:
                if "id" in response:
                    return response["id"]
                raise RuntimeError(f"Unexpected upload response: {response}")
        except Exception as exc:   # noqa: BLE001 — retriable classification done inside
            status_code = getattr(getattr(exc, "resp", None), "status", None)
            is_retriable = status_code in RETRIABLE_STATUS_CODES
            if not is_retriable:
                raise
            retry += 1
            if retry > MAX_RETRIES:
                raise RuntimeError("MAX_RETRIES exceeded")
            time.sleep(random.random() * (2 ** retry))


def publish(
    youtube: YouTubeClient,
    plan: dict[str, Any],
    video_path: Path,
    thumbnail_path: Path,
    channel_id: str,
) -> str:
    """End-to-end publish. Returns videoId."""
    assert_can_publish()
    assert_in_window()

    # Inject production_metadata into description before body build.
    pm_in = plan.get("production_metadata", {})
    meta = {
        "script_seed": pm_in.get("script_seed", ""),
        "assets_origin": pm_in.get("assets_origin", ""),
        "pipeline_version": PIPELINE_VERSION,
        "checksum": compute_checksum(video_path),
    }
    if "snippet" not in plan:
        plan["snippet"] = {}
    plan["snippet"]["description"] = inject_into_description(
        plan["snippet"].get("description", ""), meta,
    )

    body = build_insert_body(plan)
    # MediaFileUpload lazy imported — tests pass a stub media_body via plan['_media_body'] hook.
    media_body = plan.get("_media_body")
    if media_body is None:
        from googleapiclient.http import MediaFileUpload   # lazy import for test isolation
        media_body = MediaFileUpload(str(video_path), chunksize=-1, resumable=True,
                                     mimetype="video/mp4")
    insert_req = youtube.videos().insert(
        part="snippet,status",
        body=body,
        media_body=media_body,
    )
    video_id = resumable_upload(insert_req)

    # Thumbnail set
    thumb_body = plan.get("_thumb_body")
    if thumb_body is None:
        from googleapiclient.http import MediaFileUpload
        thumb_body = MediaFileUpload(str(thumbnail_path), mimetype="image/png")
    youtube.thumbnails().set(videoId=video_id, media_body=thumb_body).execute()

    # Pinned comment (D-09). end_screen_subscribe_cta is NOT an API — funnel intent only.
    pin_text = plan.get("funnel", {}).get("pinned_comment", "구독해주세요!")
    youtube.commentThreads().insert(
        part="snippet",
        body={
            "snippet": {
                "channelId": channel_id,
                "videoId": video_id,
                "topLevelComment": {"snippet": {"textOriginal": pin_text}},
            },
        },
    ).execute()

    record_upload()
    return video_id
```
