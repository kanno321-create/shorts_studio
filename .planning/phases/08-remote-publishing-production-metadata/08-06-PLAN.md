## scripts/publisher/smoke_test.py contract (D-10 + D-11)

```python
"""Phase 8 Wave 5 smoke test — one-shot real YouTube upload + cleanup.

⚠️ USER APPROVAL REQUIRED. Per CONTEXT D-11 + User Option A:
- Uploads tests/phase08/fixtures/sample_shorts.mp4 with privacyStatus="unlisted"
- Waits 30s for processingStatus == "succeeded" (Pattern: Research Area 1)
- Calls videos.delete(id=<vid>).execute() — channel history cleanup
- Raises SmokeTestCleanupFailure if cleanup non-2xx (manual delete via Studio required)

Run once to prove end-to-end chain: OAuth → build_insert_body → publish → delete.

Usage:
    python scripts/publisher/smoke_test.py --privacy=unlisted --cleanup
    python scripts/publisher/smoke_test.py --dry-run    # uses MockYouTube
"""
from __future__ import annotations

import argparse
import sys
import time
from pathlib import Path

from scripts.publisher.exceptions import SmokeTestCleanupFailure
from scripts.publisher.oauth import get_credentials, CLIENT_SECRET_PATH, TOKEN_PATH
from scripts.publisher.youtube_uploader import publish


SMOKE_VIDEO_PATH = Path("tests/phase08/fixtures/sample_shorts.mp4")
PROCESSING_WAIT_SECONDS = 30


def _build_smoke_plan() -> dict:
    """Minimal plan for smoke test — privacy=unlisted hardcoded."""
    return {
        "snippet": {
            "title": f"SMOKE_TEST_{int(time.time())}",
            "description": "Phase 8 Wave 5 automated smoke test. Delete immediately.",
            "tags": ["smoke", "test"],
            "categoryId": "24",
            "defaultLanguage": "ko",
        },
        "status": {"privacyStatus": "unlisted",   # <-- HARDCODED unlisted (not public)
                   "embeddable": False, "publicStatsViewable": False},
        "ai_disclosure": {"syntheticMedia": True},
        "production_metadata": {
            "script_seed": "smoke_test",
            "assets_origin": "smoke:test",
        },
        "funnel": {"pinned_comment": "smoke test — delete immediately"},
    }


def _wait_for_processing(youtube, video_id: str, *, seconds: int = PROCESSING_WAIT_SECONDS) -> None:
    """Poll videos.list until processingStatus=='succeeded' or timeout."""
    deadline = time.time() + seconds
    while time.time() < deadline:
        resp = youtube.videos().list(part="status", id=video_id).execute()
        items = resp.get("items", [])
        if items and items[0]["status"].get("processingStatus") == "succeeded":
            return
        time.sleep(2)
    # Timeout: proceed to delete anyway — cleanup must happen.


def _delete_video(youtube, video_id: str) -> None:
    """videos.delete with error → SmokeTestCleanupFailure."""
    try:
        youtube.videos().delete(id=video_id).execute()
    except Exception as exc:
        raise SmokeTestCleanupFailure(video_id, str(exc)) from exc


def run_smoke_test(*, youtube=None, channel_id: str = "UCsmoke",
                   privacy: str = "unlisted", cleanup: bool = True) -> str:
    """Returns the uploaded video_id post-cleanup.

    If `youtube` is None, builds real client via get_credentials() → build("youtube","v3",creds).
    Tests pass MockYouTube instance directly to bypass OAuth.
    """
    if privacy != "unlisted":
        raise ValueError("Smoke test MUST use privacyStatus='unlisted' (D-11 safety)")
    if youtube is None:
        # Precondition check (BLOCKING if missing — RESEARCH Environment Availability)
        if not CLIENT_SECRET_PATH.exists():
            raise FileNotFoundError(
                f"{CLIENT_SECRET_PATH} missing — 대표님 Google Cloud Console OAuth 2.0 Desktop client 생성 필요"
            )
        creds = get_credentials()
        from googleapiclient.discovery import build   # lazy import
        youtube = build("youtube", "v3", credentials=creds)
    plan = _build_smoke_plan()
    video_id = publish(
        youtube, plan,
        SMOKE_VIDEO_PATH, SMOKE_VIDEO_PATH,   # fixture is 1-byte .mp4; thumbnail path reused
        channel_id,
    )
    if cleanup:
        _wait_for_processing(youtube, video_id)
        _delete_video(youtube, video_id)
    return video_id


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--privacy", default="unlisted",
                        choices=["unlisted"],   # literally the only allowed value
                        help="privacyStatus (unlisted only — D-11 safety)")
    parser.add_argument("--cleanup", action="store_true", default=True)
    parser.add_argument("--dry-run", action="store_true",
                        help="Use MockYouTube instead of real Data API v3")
    args = parser.parse_args(argv)

    if args.dry_run:
        sys.path.insert(0, str(Path("tests/phase08").resolve()))
        from mocks.youtube_mock import MockYouTube   # noqa: E402
        yt = MockYouTube()
    else:
        yt = None

    vid = run_smoke_test(youtube=yt, privacy=args.privacy, cleanup=args.cleanup)
    print(f"SMOKE_VIDEO_ID: {vid}")
    print("SMOKE_STATUS: cleanup-complete" if args.cleanup else "SMOKE_STATUS: upload-only")
    return 0


if __name__ == "__main__":
    if hasattr(sys.stdout, "reconfigure"):
        sys.stdout.reconfigure(encoding="utf-8")   # Windows cp949 guard
    sys.exit(main())
```
