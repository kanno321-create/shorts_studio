## scripts/publisher/publish_lock.py contract (PUB-03 D-06)

```python
"""48h+ jitter filesystem lock — Phase 8 PUB-03 enforcement.

Pattern: Phase 5 scripts/orchestrator/checkpointer.py atomic os.replace.
D-06: .planning/publish_lock.json (gitignored). SHORTS_PUBLISH_LOCK_PATH env var override for tests.
"""
from __future__ import annotations

import json
import os
import random
from datetime import datetime, timezone
from pathlib import Path

from scripts.publisher.exceptions import PublishLockViolation

DEFAULT_LOCK_PATH = Path(".planning/publish_lock.json")
MIN_ELAPSED_HOURS = 48
MAX_JITTER_MIN = 720   # 0~12h additional jitter → effective 48~60h


def _lock_path() -> Path:
    return Path(os.environ.get("SHORTS_PUBLISH_LOCK_PATH",
                               str(DEFAULT_LOCK_PATH)))


def assert_can_publish(*, now_utc: datetime | None = None) -> None:
    """Raises PublishLockViolation if elapsed_min < MIN_ELAPSED_HOURS*60 + jitter."""
    path = _lock_path()
    if not path.exists():
        return   # first-ever upload — no lock
    data = json.loads(path.read_text(encoding="utf-8"))
    last = datetime.fromisoformat(data["last_upload_iso"])
    jitter = int(data.get("jitter_applied_min", 0))
    required_min = MIN_ELAPSED_HOURS * 60 + jitter
    now = now_utc or datetime.now(timezone.utc)
    elapsed_min = (now - last).total_seconds() / 60
    if elapsed_min < required_min:
        raise PublishLockViolation(int(required_min - elapsed_min))


def record_upload(*, now_utc: datetime | None = None,
                  jitter_min: int | None = None) -> None:
    """Atomic write: tmp file + os.replace."""
    path = _lock_path()
    path.parent.mkdir(parents=True, exist_ok=True)
    now = now_utc or datetime.now(timezone.utc)
    jitter = jitter_min if jitter_min is not None else random.randint(0, MAX_JITTER_MIN)
    payload = {
        "last_upload_iso": now.isoformat(),
        "jitter_applied_min": jitter,
        "_schema": 1,
    }
    tmp = path.with_suffix(".tmp")
    tmp.write_text(json.dumps(payload, ensure_ascii=False, indent=2),
                   encoding="utf-8")
    os.replace(tmp, path)   # atomic Windows + Unix
```

```python
"""KST peak window enforcement — Phase 8 PUB-03 D-07.

Weekday (Mon-Fri, 0..4): 20:00-22:59 KST (hour < 23 exclusive upper).
Weekend (Sat-Sun, 5..6): 12:00-14:59 KST (hour < 15 exclusive upper).
"""
from __future__ import annotations

from datetime import datetime
from zoneinfo import ZoneInfo

from scripts.publisher.exceptions import PublishWindowViolation

KST = ZoneInfo("Asia/Seoul")


def assert_in_window(*, now_kst: datetime | None = None) -> None:
    now = now_kst or datetime.now(KST)
    weekday = now.weekday()
    hour = now.hour
    is_weekday = weekday < 5
    if is_weekday:
        ok = 20 <= hour < 23
    else:
        ok = 12 <= hour < 15
    if not ok:
        raise PublishWindowViolation(weekday=weekday, hour=hour)
```
