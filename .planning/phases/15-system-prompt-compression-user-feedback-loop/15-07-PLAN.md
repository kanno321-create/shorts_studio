       ```python
       """Phase 15 acceptance — SPC-01~06 + UFL-01~04 aggregator.
       
       Phase 13/14 패턴 승계. 10 SC aggregate → exit 0 (all green) or exit 1 (any fail).
       """
       from __future__ import annotations
       import glob
       import json
       import re
       import subprocess
       import sys
       from pathlib import Path
       
       REPO = Path(".")
       EVIDENCE = REPO / ".planning/phases/15-system-prompt-compression-user-feedback-loop/evidence"
       
       
       def check_spc01_invokers_patch() -> tuple[bool, str, str]:
           inv = (REPO / "scripts/orchestrator/invokers.py").read_text(encoding="utf-8")
           has_tempfile = "tempfile.NamedTemporaryFile" in inv
           has_flag = "--append-system-prompt-file" in inv
           has_unlink = "os.unlink" in inv
           passed = has_tempfile and has_flag and has_unlink
           return passed, "SC-SPC-01 invokers.py tempfile patch", (
               f"tempfile={has_tempfile} flag={has_flag} unlink={has_unlink}"
           )
       
       
       def check_spc02_supervisor_compressed() -> tuple[bool, str, str]:
           agent = (REPO / ".claude/agents/supervisor/shorts-supervisor/AGENT.md").read_text(encoding="utf-8")
           body = agent.split("---", 2)[-1].strip()
           size_ok = len(body) < 7000
           ref1 = (REPO / ".claude/agents/supervisor/shorts-supervisor/references/supervisor_variant.md").exists()
           ref2 = (REPO / ".claude/agents/supervisor/shorts-supervisor/references/inspector_paths.md").exists()
           must_remember = "MUST REMEMBER" in agent
           # 5-block schema via verify_agent_md_schema.py
           schema_ok = subprocess.run(
               ["py", "-3.11", "scripts/validate/verify_agent_md_schema.py", "--fail-on-drift"],
               capture_output=True, text=True, encoding="utf-8", errors="replace",
           ).returncode == 0
           passed = size_ok and ref1 and ref2 and must_remember and schema_ok
           return passed, "SC-SPC-02 supervisor AGENT.md 압축", (
               f"body={len(body)} refs={ref1 and ref2} MR={must_remember} schema={schema_ok}"
           )
       
       
       def check_spc03_size_ceiling() -> tuple[bool, str, str]:
           rc = subprocess.run(
               ["py", "-3.11", "scripts/validate/verify_agent_md_size.py", "--ceiling", "18000"],
               capture_output=True, text=True, encoding="utf-8", errors="replace",
           ).returncode
           return rc == 0, "SC-SPC-03 AGENT.md 18000 ceiling", f"exit={rc}"
       
       
       def check_spc04_cli_flag_logged() -> tuple[bool, str, str]:
           log = EVIDENCE / "15-01-cli-probe.log"
           if not log.exists():
               return False, "SC-SPC-04 CLI flag evidence", "log 파일 없음"
           text = log.read_text(encoding="utf-8", errors="replace")
           has_found = ("file not found" in text.lower()
                        or "파일" in text
                        or "enoent" in text.lower())
           no_unknown = "unknown option" not in text.lower()
           passed = has_found and no_unknown
           return passed, "SC-SPC-04 CLI flag 실측", (
               f"has_found={has_found} no_unknown={no_unknown}"
           )
       
       
       def check_spc05_contract_tests() -> tuple[bool, str, str]:
           result = subprocess.run(
               ["py", "-3.11", "-m", "pytest",
                "tests/adapters/test_invokers_encoding_contract.py",
                "--no-cov", "-q", "--tb=no"],
               capture_output=True, text=True, encoding="utf-8", errors="replace",
           )
           m = re.search(r"(\d+) passed", result.stdout)
           passed_count = int(m.group(1)) if m else 0
           return passed_count >= 10, "SC-SPC-05 invokers encoding contract", (
               f"passed={passed_count} rc={result.returncode}"
           )
       
       
       def check_spc06_live_smoke() -> tuple[bool, str, str]:
           e2e_files = list(EVIDENCE.glob("smoke_e2e_*.json"))
           if not e2e_files:
               return False, "SC-SPC-06 live smoke evidence", "smoke_e2e_*.json 없음"
           data = json.loads(e2e_files[0].read_text(encoding="utf-8"))
           conditions = {
               "dispatched=13": data.get("dispatched_count") == 13,
               "rc1_count=0": data.get("supervisor_rc1_count") == 0,
               "video_id": bool(data.get("final_video_id")),
               "cost<=5": (data.get("total_cost_usd") or 0) <= 5.00,
           }
           passed = all(conditions.values())
           return passed, "SC-SPC-06 live smoke retry", (
               " ".join(f"{k}={v}" for k, v in conditions.items())
           )
       
       
       def check_ufl01_revision() -> tuple[bool, str, str]:
           smoke = (REPO / "scripts/smoke/phase13_live_smoke.py").read_text(encoding="utf-8")
           inv = (REPO / "scripts/orchestrator/invokers.py").read_text(encoding="utf-8")
           passed = (
               "--revision-from" in smoke
               and "def _apply_revision" in smoke
               and "prior_user_feedback" in inv
           )
           return passed, "SC-UFL-01 revision-from + feedback", (
               f"flag_in_smoke=True feedback_in_invokers={'prior_user_feedback' in inv}"
           )
       
       
       def check_ufl02_prescripted() -> tuple[bool, str, str]:
           smoke = (REPO / "scripts/smoke/phase13_live_smoke.py").read_text(encoding="utf-8")
           passed = (
               "class _PreScriptedProducerInvoker" in smoke
               and "--revise-script" in smoke
           )
           return passed, "SC-UFL-02 revise-script", f"smoke_has_both={passed}"
       
       
       def check_ufl03_pause() -> tuple[bool, str, str]:
           pipeline = (REPO / "scripts/orchestrator/shorts_pipeline.py").read_text(encoding="utf-8")
           smoke = (REPO / "scripts/smoke/phase13_live_smoke.py").read_text(encoding="utf-8")
           passed = (
               "class PipelinePauseSignal" in pipeline
               and "--pause-after" in smoke
               and "_handle_pause_signal" in smoke
           )
           return passed, "SC-UFL-03 pause-after signal", f"all_present={passed}"
       
       
       def check_ufl04_rating_cli() -> tuple[bool, str, str]:
           rate = (REPO / "scripts/smoke/rate_video.py").exists()
           verify = (REPO / "scripts/validate/verify_feedback_format.py").exists()
           mem = (REPO / ".claude/memory/feedback_video_quality.md").exists()
           passed = rate and verify and mem
           return passed, "SC-UFL-04 rating CLI", (
               f"rate={rate} verify={verify} mem={mem}"
           )
       
       
       CHECKS = [
           check_spc01_invokers_patch,
           check_spc02_supervisor_compressed,
           check_spc03_size_ceiling,
           check_spc04_cli_flag_logged,
           check_spc05_contract_tests,
           check_spc06_live_smoke,
           check_ufl01_revision,
           check_ufl02_prescripted,
           check_ufl03_pause,
           check_ufl04_rating_cli,
       ]
       
       
       def main() -> int:
           try:
               sys.stdout.reconfigure(encoding="utf-8")
           except (AttributeError, OSError):
               pass
           print("=" * 70)
           print("Phase 15 Acceptance Gate (대표님) — 10 SC")
           print("=" * 70)
           results = []
           for fn in CHECKS:
               try:
                   passed, label, detail = fn()
               except Exception as err:
                   passed, label, detail = False, fn.__name__, f"ERROR: {err}"
               results.append((passed, label, detail))
               mark = "✅" if passed else "❌"
               print(f"{mark}  {label:45s} {detail}")
           fails = [r for r in results if not r[0]]
           print("=" * 70)
           if fails:
               print(f"[phase15-acceptance] {len(fails)}/10 SC 실패 (대표님).")
               return 1
           print(f"[phase15-acceptance] 10/10 SC 전수 통과 (대표님).")
           return 0
       
       
       if __name__ == "__main__":
           raise SystemExit(main())
       ```
