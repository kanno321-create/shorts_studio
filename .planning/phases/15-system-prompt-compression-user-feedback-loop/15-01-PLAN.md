       """SPC-04 — Claude CLI `--append-system-prompt-file` flag 실측 확증.

       """
       from __future__ import annotations
       import shutil
       import subprocess
       from pathlib import Path
       import pytest
       
       EVIDENCE = Path(
           ".planning/phases/15-system-prompt-compression-user-feedback-loop/evidence/15-01-cli-probe.log"
       )
       
       @pytest.fixture(scope="module")
       def claude_bin() -> str:
           bin_path = shutil.which("claude")
           if not bin_path:
               pytest.skip("claude CLI 미설치 — Max 구독 환경 아님 (대표님)")
           return bin_path
       
       def _append_evidence(tag: str, stdout: str, stderr: str, rc: int) -> None:
           EVIDENCE.parent.mkdir(parents=True, exist_ok=True)
           with EVIDENCE.open("a", encoding="utf-8") as fp:
               fp.write(f"\n## {tag}\nrc={rc}\n--- stdout ---\n{stdout}\n--- stderr ---\n{stderr}\n")
       
       def test_claude_binary_resolvable(claude_bin):
           """shutil.which 가 claude 바이너리 절대경로를 반환합니다."""
           assert Path(claude_bin).exists()
       
       def test_append_system_prompt_file_flag_recognized(claude_bin):
           """`--append-system-prompt-file` 이 CLI 에 인식됨 (file-not-found error = flag 인식 증거)."""
           result = subprocess.run(
               [claude_bin, "--print", "--append-system-prompt-file",
                "/tmp/nonexistent-15-01-probe"],
               input="hi", capture_output=True, text=True,
               encoding="utf-8", errors="replace", timeout=15,
           )
           _append_evidence("test_append_system_prompt_file_flag", result.stdout, result.stderr, result.returncode)
           # flag 인식 시: rc != 0 + stderr 에 file-related error
           assert result.returncode != 0, "flag 인식 시 nonexistent file → 비영 rc 필수"
           assert ("not found" in result.stderr.lower()
                   or "파일" in result.stderr
                   or "enoent" in result.stderr.lower()), (
               f"flag 인식 안 됨 — stderr: {result.stderr[:500]} (대표님)"
           )
           # unknown option error 가 아님 확인
           assert "unknown option" not in result.stderr.lower()
           assert "unrecognized" not in result.stderr.lower()
       
       def test_system_prompt_file_flag_also_exists(claude_bin):
           """`--system-prompt-file` (non-append variant) 도 인식 (SPC-04 Option B 문서화)."""
           result = subprocess.run(
               [claude_bin, "--print", "--system-prompt-file",
                "/tmp/nonexistent-15-01-probe-sys"],

               encoding="utf-8", errors="replace", timeout=15,
           )
           _append_evidence("test_system_prompt_file_flag", result.stdout, result.stderr, result.returncode)
           assert result.returncode != 0
           assert "unknown option" not in result.stderr.lower()
