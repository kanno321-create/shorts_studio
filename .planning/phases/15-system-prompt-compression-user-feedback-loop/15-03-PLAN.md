       ```python
       """SPC-03 — AGENT.md body char ceiling enforcement.
       
       Phase 12 AGENT-STD-01 5-block invariant 보완. drift 방지용 soft ceiling.
       기본 CHAR_LIMIT=18000 (현재 max=17426 + 3% 여유) — 현 시점 0 breaking change.
       """
       from __future__ import annotations
       import argparse
       import json
       import sys
       from pathlib import Path
       
       CHAR_LIMIT = 18000
       
       
       def strip_frontmatter(text: str) -> str:
           if text.startswith("---"):
               parts = text.split("---", 2)
               if len(parts) >= 3:
                   return parts[2]
           return text
       
       
       def scan_agents(root: Path) -> dict[str, int]:
           results: dict[str, int] = {}
           for pattern in ("producers/*/AGENT.md", "supervisor/*/AGENT.md"):
               for agent_md in root.glob(pattern):
                   body = strip_frontmatter(
                       agent_md.read_text(encoding="utf-8")
                   ).strip()
                   key = str(agent_md.relative_to(root))
                   results[key] = len(body)
           return results
       
       
       def main(argv: list[str] | None = None) -> int:
           parser = argparse.ArgumentParser(description="SPC-03 AGENT.md size ceiling")
           parser.add_argument("--ceiling", type=int, default=CHAR_LIMIT)
           parser.add_argument("--agents-root", type=Path, default=Path(".claude/agents"))
           parser.add_argument("--json-out", type=Path, default=None)
           args = parser.parse_args(argv)
       
           # Windows cp949 guard
           try:
               sys.stdout.reconfigure(encoding="utf-8")
           except (AttributeError, OSError):
               pass
       
           sizes = scan_agents(args.agents_root)
           violators = {k: v for k, v in sizes.items() if v > args.ceiling}
       
           if args.json_out:
               args.json_out.write_text(
                   json.dumps({"ceiling": args.ceiling, "sizes": sizes, "violators": violators}, ensure_ascii=False, indent=2),
                   encoding="utf-8",
               )
       
           if violators:
               print(f"[SPC-03] CHAR_LIMIT={args.ceiling} 초과 agent (대표님):")
               for name, size in sorted(violators.items()):
                   print(f"  - {name}: {size} chars (+{size - args.ceiling})")
               return 1
       
           print(f"[SPC-03] 전체 {len(sizes)} AGENT.md 모두 {args.ceiling} chars 이하 (대표님).")
           return 0
       
       
       if __name__ == "__main__":
           raise SystemExit(main())
       ```
