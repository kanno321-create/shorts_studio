       """UFL-04 — 대표님 영상 품질 평가 CLI.

       사용 예:
           py -3.11 scripts/smoke/rate_video.py \
               --video-id dQw4w9WgXcQ \
               --rating 3 \
               --feedback "조명이 어두움, 음성 톤이 단조로움"

       저장: .claude/memory/feedback_video_quality.md (D-2 Lock 예외 — 대표님 직접 입력).
       """
       from __future__ import annotations
       import argparse
       import re
       import sys
       import time
       from collections import Counter
       from pathlib import Path

       DEFAULT_MEMORY = Path(".claude/memory/feedback_video_quality.md")

       SEED_FRONTMATTER = """---
       d2_exception: true
       reason: "대표님 직접 입력 — F-D2-EXCEPTION 패턴"
       phase_added: 15-system-prompt-compression-user-feedback-loop
       created: {today}
       purpose: 대표님 영상 품질 subjective rating 집계
       ---

       # 영상 품질 피드백 로그

       ---
       """

       NOUN_REGEX = re.compile(r"[가-힣]{2,}")
       STOPWORDS = {"그리고", "하지만", "그래서", "아주", "매우", "많이", "조금"}


       def extract_keywords(text: str, top_n: int = 3) -> list[str]:
           tokens = [t for t in NOUN_REGEX.findall(text) if t not in STOPWORDS]
           freq = Counter(tokens)
           return [w for w, _ in freq.most_common(top_n)]


       def format_entry(
           *, today: str, video_id: str, session_id: str | None,
           niche: str | None, rating: int, feedback: str,
       ) -> str:
           keywords = extract_keywords(feedback)
           lines = [
               f"## {today} {video_id}",
               f"- session_id: {session_id or '(미지정)'}",
               f"- niche: {niche or '(미지정)'}",
               f"- rating: {rating}/5",
               f"- feedback: {feedback}",
               f"- keywords: {keywords}",
               "",
           ]
           return "\n".join(lines) + "\n"


       def ensure_seed(memory_path: Path, today: str) -> None:
           if not memory_path.exists():
               memory_path.parent.mkdir(parents=True, exist_ok=True)
               memory_path.write_text(
                   SEED_FRONTMATTER.format(today=today),
                   encoding="utf-8",
               )


       def append_rating(
           *, memory_path: Path, video_id: str, rating: int, feedback: str,
           session_id: str | None = None, niche: str | None = None,
       ) -> None:
           if not isinstance(rating, int) or rating < 1 or rating > 5:
               raise ValueError(
                   f"rating 은 1~5 사이 정수여야 합니다 (대표님): {rating!r}"
               )
           if not video_id:
               raise ValueError("video_id 는 필수입니다 (대표님).")
           if not feedback:
               raise ValueError("feedback 은 필수입니다 (대표님).")

           today = time.strftime("%Y-%m-%d")
           ensure_seed(memory_path, today)
           entry = format_entry(
               today=today, video_id=video_id, session_id=session_id,
               niche=niche, rating=rating, feedback=feedback,
           )
           with memory_path.open("a", encoding="utf-8") as fp:
               fp.write(entry)


       def main(argv: list[str] | None = None) -> int:
           parser = argparse.ArgumentParser(
               description="UFL-04 대표님 영상 품질 rating CLI"
           )
           parser.add_argument("--video-id", required=True)
           parser.add_argument("--rating", type=int, required=True)
           parser.add_argument("--feedback", required=True)
           parser.add_argument("--session-id", default=None)
           parser.add_argument("--niche", default=None)
           parser.add_argument("--memory-path", type=Path, default=DEFAULT_MEMORY)
           args = parser.parse_args(argv)

           try:
               sys.stdout.reconfigure(encoding="utf-8")
           except (AttributeError, OSError):
               pass

           append_rating(
               memory_path=args.memory_path,
               video_id=args.video_id, rating=args.rating,
               feedback=args.feedback, session_id=args.session_id,
               niche=args.niche,
           )
           print(
               f"[rate_video] 기록 완료 (대표님): {args.video_id} "
               f"rating={args.rating}/5 → {args.memory_path}"
           )
           return 0


       if __name__ == "__main__":
           raise SystemExit(main())
       ```

       """UFL-04 — feedback_video_quality.md format validator.

       Markdown H2 + 4 required fields (session_id, niche, rating, feedback).
       rating 은 1-5/5 format.
       """
       from __future__ import annotations
       import argparse
       import re
       import sys
       from pathlib import Path

       DEFAULT_PATH = Path(".claude/memory/feedback_video_quality.md")

       H2_RE = re.compile(r"^## \d{4}-\d{2}-\d{2} \S+\s*$")
       FIELD_RE = re.compile(r"^-\s*(\w+)\s*:\s*(.+)$")
       RATING_RE = re.compile(r"^[1-5]/5$")

       REQUIRED_FIELDS = {"session_id", "niche", "rating", "feedback"}


       def parse_entries(text: str) -> list[dict]:
           entries: list[dict] = []
           current: dict | None = None
           for i, line in enumerate(text.splitlines(), start=1):
               if H2_RE.match(line):
                   if current:
                       entries.append(current)
                   current = {"_line": i, "_header": line, "_fields": {}}
               elif current is not None:
                   m = FIELD_RE.match(line)
                   if m:
                       current["_fields"][m.group(1)] = m.group(2).strip()
           if current:
               entries.append(current)
           return entries


       def validate_entry(entry: dict) -> list[str]:
           errors: list[str] = []
           fields = entry["_fields"]
           missing = REQUIRED_FIELDS - set(fields.keys())
           for m in sorted(missing):
               errors.append(f"line {entry['_line']}: 누락 field '{m}' (대표님)")
           rating = fields.get("rating", "")
           if rating and not RATING_RE.match(rating):
               errors.append(
                   f"line {entry['_line']}: rating format 위반 '{rating}' — 1/5~5/5 요구 (대표님)"
               )
           return errors


       def main(argv: list[str] | None = None) -> int:
           parser = argparse.ArgumentParser(description="UFL-04 format validator")
           parser.add_argument("--path", type=Path, default=DEFAULT_PATH)
           args = parser.parse_args(argv)

           try:
               sys.stdout.reconfigure(encoding="utf-8")
           except (AttributeError, OSError):
               pass

           if not args.path.exists():
               print(f"[feedback-fmt] 파일 없음 (대표님): {args.path}")
               return 1

           text = args.path.read_text(encoding="utf-8")
           entries = parse_entries(text)
           all_errors: list[str] = []
           for e in entries:
               all_errors.extend(validate_entry(e))

           if all_errors:
               print(f"[feedback-fmt] 위반 {len(all_errors)} 건 (대표님):")
               for err in all_errors:
                   print(f"  - {err}")
               return 1

           print(f"[feedback-fmt] 전체 {len(entries)} entry 검증 완료 (대표님).")
           return 0


       if __name__ == "__main__":
           raise SystemExit(main())
       ```
