```python
import pathlib, re, pytest
from scripts.validate.parse_frontmatter import parse_frontmatter

STRUCTURAL_DIR = pathlib.Path(".claude/agents/inspectors/structural")

@pytest.fixture(scope="module")
def structural_agents():
    files = list(STRUCTURAL_DIR.rglob("AGENT.md"))
    return [(f, *parse_frontmatter(f)) for f in files]

def test_exactly_3_structural_inspectors(structural_agents):
    assert len(structural_agents) == 3

@pytest.mark.parametrize("path,meta,body", [...])  # fill in from fixture
def test_frontmatter_role_category_maxturns(path, meta, body):
    assert meta["role"] == "inspector"
    assert meta["category"] == "structural"
    assert int(meta["maxTurns"]) == 1

def test_logicqa_block_present(structural_agents):
    for path, meta, body in structural_agents:
        assert "<main_q>" in body, f"{path}: missing main_q"
        assert "<sub_qs>" in body, f"{path}: missing sub_qs"
        # count 5 sub-questions (q1..q5)
        for q_id in ["q1", "q2", "q3", "q4", "q5"]:
            assert re.search(rf"\b{q_id}\s*:", body), f"{path}: missing {q_id}"

def test_must_remember_contains_rub_violations(structural_agents):
    for path, meta, body in structural_agents:
        # Extract MUST REMEMBER section
        mr_match = re.search(r"## MUST REMEMBER.*?(?=\n##|\Z)", body, re.DOTALL)
        assert mr_match, f"{path}: no MUST REMEMBER section"
        mr = mr_match.group(0)
        assert "창작 금지" in mr, f"{path}: MUST REMEMBER missing '창작 금지' (RUB-02)"
        assert "producer_prompt" in mr, f"{path}: MUST REMEMBER missing producer_prompt reference (RUB-06)"

def test_inputs_no_producer_prompt(structural_agents):
    for path, meta, body in structural_agents:
        # Extract Inputs section
        inputs_match = re.search(r"## Inputs.*?(?=\n##)", body, re.DOTALL)
        if inputs_match:
            inputs = inputs_match.group(0)
            # Inputs table should NOT list producer_prompt as a Flag
            # (it's OK if the word appears in the RUB-06 note "producer_prompt 미포함")
            # Check: no table row with "producer_prompt" or "producer_system_context" Flag
            assert not re.search(r"\|\s*producer_prompt\s*\|", inputs)
            assert not re.search(r"\|\s*producer_system_context\s*\|", inputs)

def test_ins_schema_integrity_content_05(structural_agents):
    for path, meta, body in structural_agents:
        if meta["name"] == "ins-schema-integrity":
            assert "9:16" in body
            assert "1080×1920" in body or "1080x1920" in body
            assert "59" in body
            return
    pytest.fail("ins-schema-integrity not found")
```
