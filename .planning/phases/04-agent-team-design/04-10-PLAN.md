**File 1: scripts/validate/grep_gan_contamination.py**
```python
"""RUB-06 GAN separation enforcement.

Inspector AGENT.md must NOT have producer_prompt / producer_system_context
as Flag entries in Inputs table (these would leak Producer context).

It IS OK for these strings to appear in MUST REMEMBER / documentation 
as negation ("producer_prompt 읽기 금지").

Strategy: parse each Inspector AGENT.md, extract ## Inputs section, check
table rows for forbidden flag names.
"""
import pathlib, re, sys

INSPECTORS_DIR = pathlib.Path(".claude/agents/inspectors")
FORBIDDEN_FLAGS = ["producer_prompt", "producer_system_context", "producer_system"]

def check_file(md_path):
    text = md_path.read_text("utf-8")
    # Find ## Inputs section (between ## Inputs and next ##)
    m = re.search(r"## Inputs\s*\n(.*?)(?=\n##\s|\Z)", text, re.DOTALL)
    if not m:
        return []
    inputs_block = m.group(1)
    violations = []
    for flag in FORBIDDEN_FLAGS:
        # Check for Markdown table row starting with | flag | or | --flag |
        pattern = rf"\|\s*-?-?{re.escape(flag)}\s*\|"
        if re.search(pattern, inputs_block):
            violations.append(f"{md_path}: Inputs table contains forbidden flag '{flag}'")
    return violations

def main():
    all_violations = []
    for md_path in INSPECTORS_DIR.rglob("AGENT.md"):
        all_violations.extend(check_file(md_path))
    if all_violations:
        for v in all_violations:
            print(v, file=sys.stderr)
        print(f"GAN_CONTAMINATION: {len(all_violations)} violation(s)", file=sys.stderr)
        sys.exit(1)
    count = sum(1 for _ in INSPECTORS_DIR.rglob("AGENT.md"))
    print(f"GAN_CLEAN: {count} inspector AGENT.md files verified clean (RUB-06)")

if __name__ == "__main__":
    main()
```

```python
import pathlib, re
import pytest
from scripts.validate.parse_frontmatter import parse_frontmatter

INSPECTORS_DIR = pathlib.Path(".claude/agents/inspectors")

INSPECTOR_PATHS = list(INSPECTORS_DIR.rglob("AGENT.md"))

def test_exactly_17_inspectors():
    assert len(INSPECTOR_PATHS) == 17, f"Expected 17 inspectors, got {len(INSPECTOR_PATHS)}"

@pytest.mark.parametrize("md_path", INSPECTOR_PATHS)
def test_logicqa_block_has_exactly_5_subqs(md_path):
    _, body = parse_frontmatter(md_path)
    # Find <sub_qs>...</sub_qs> block
    m = re.search(r"<sub_qs>(.*?)</sub_qs>", body, re.DOTALL)
    assert m, f"{md_path}: missing <sub_qs> block"
    sub_qs_content = m.group(1)
    # Count q1..q5 tokens (anchored: "q1:", "q2:", ...)
    sub_count = sum(1 for qid in ["q1", "q2", "q3", "q4", "q5"] if re.search(rf"\b{qid}\s*:", sub_qs_content))
    assert sub_count == 5, f"{md_path}: expected exactly 5 sub_qs, found {sub_count}"
    # Guard against q6 or extras
    assert not re.search(r"\bq6\s*:", sub_qs_content), f"{md_path}: unexpected q6"

@pytest.mark.parametrize("md_path", INSPECTOR_PATHS)
def test_main_q_block_present(md_path):
    _, body = parse_frontmatter(md_path)
    assert "<main_q>" in body, f"{md_path}: missing <main_q>"
    assert "</main_q>" in body, f"{md_path}: missing </main_q>"
```
