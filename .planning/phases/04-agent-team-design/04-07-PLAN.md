**test_af5_real_face_block.py:**
```python
import json, pathlib, re
import pytest

AF_BANK = pathlib.Path(".claude/agents/_shared/af_bank.json")

# Mirror ins-mosaic domain + keyword regex
KOREAN_PRESS_DOMAINS = re.compile(
    r"(chosun\.com|joongang\.co\.kr|donga\.com|hani\.co\.kr|"
    r"mbc\.co\.kr|kbs\.co\.kr|sbs\.co\.kr|jtbc\.co\.kr|"
    r"news\.naver\.com|news\.daum\.net)",
    re.IGNORECASE
)
REAL_FACE_KEYWORDS = re.compile(
    r"(news|victim|press-photo|press_photo|실존|피해자)",
    re.IGNORECASE
)

def af5_blocked(entry):
    url = entry.get("url_pattern", "")
    caption = entry.get("caption", "")
    full_text = f"{url} {caption}"
    if KOREAN_PRESS_DOMAINS.search(full_text):
        return True
    if REAL_FACE_KEYWORDS.search(full_text):
        return True
    return False

@pytest.fixture(scope="module")
def af_bank():
    return json.loads(AF_BANK.read_text("utf-8"))

def test_af5_real_face_100pct_block(af_bank):
    af5 = af_bank["af5_real_face"]
    assert len(af5) >= 10
    blocked = sum(1 for entry in af5 if af5_blocked(entry))
    assert blocked == len(af5), f"AF-5 block rate {blocked}/{len(af5)} != 100%"
```

NOTE: Plan 01 Task 2의 af_bank af5_real_face 샘플은 url_pattern 또는 caption 필드에 이 regex에 걸릴 키워드 포함 필수 (예: "press-photo.com/victim_2020" + caption="KBS 뉴스 피해자 A씨").
