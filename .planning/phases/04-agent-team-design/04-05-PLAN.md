```python
import json, pathlib, re
import pytest

AF_BANK = pathlib.Path(".claude/agents/_shared/af_bank.json")

# AF-13 K-pop regex bank (mirrors §5.5 line 1162)
KPOP_ARTISTS = re.compile(
    r"BTS|방탄소년단|블랙핑크|BLACKPINK|뉴진스|NewJeans|아이브|IVE|"
    r"에스파|aespa|르세라핌|LE SSERAFIM|스트레이 키즈|Stray Kids|"
    r"세븐틴|SEVENTEEN|NCT|트와이스|TWICE",
    re.IGNORECASE
)
KPOP_TITLES = re.compile(
    r"Dynamite|다이너마이트|Butter|Spring Day|Savage|Ditto|"
    r"Super Shy|Hype Boy|Love Dive|ETA|How Sweet",
    re.IGNORECASE
)

# AF-4: sample bank의 name 필드를 그대로 blocklist로 사용
def af4_blocked(name, bank_af4):
    name_lower = name.lower()
    for entry in bank_af4:
        if entry["name"].lower() in name_lower or name_lower in entry["name"].lower():
            return True
    return False

def af13_blocked(artist, title):
    if KPOP_ARTISTS.search(artist) or KPOP_TITLES.search(title):
        return True
    return False

@pytest.fixture(scope="module")
def af_bank():
    return json.loads(AF_BANK.read_text("utf-8"))

def test_af4_voice_clone_100pct_block(af_bank):
    af4 = af_bank["af4_voice_clone"]
    assert len(af4) >= 10
    blocked = sum(1 for entry in af4 if af4_blocked(entry["name"], af4))
    assert blocked == len(af4), f"AF-4 block rate {blocked}/{len(af4)} != 100%"

def test_af13_kpop_100pct_block(af_bank):
    af13 = af_bank["af13_kpop"]
    assert len(af13) >= 10
    blocked = sum(1 for entry in af13 if af13_blocked(entry.get("artist",""), entry.get("title","")))
    assert blocked == len(af13), f"AF-13 block rate {blocked}/{len(af13)} != 100%"
```

NOTE: Sample bank in Plan 01 Task 2 must be crafted so that artist/title fields are detectable by KPOP_ARTISTS/KPOP_TITLES regex → Plan 01 writes samples, Plan 05 verifies 100% regex coverage.
