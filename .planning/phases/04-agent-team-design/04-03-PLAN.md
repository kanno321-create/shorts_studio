**test_ins_korean_naturalness.py** — THE KEY REGRESSION TEST (VALIDATION map row 4-03-02):
```python
import json, pathlib, re
import pytest

KOREAN_SAMPLES_PATH = pathlib.Path(".claude/agents/_shared/korean_speech_samples.json")

# Regex patterns mirror §5.3 line 1127-1131
HAOCHE_ENDINGS = re.compile(r"(하오|이오|보오|가오|구려|다네|습니다|올시다)\.?\s*$")
HAEYOCHE_ENDINGS = re.compile(r"(해요|이에요|예요|지요|거든요|네요|잖아요)\.?\s*$")
BANMAL_ENDINGS = re.compile(r"(해|야|지|야지|거든|네|잖아|다)\.?\s*$")
CALL_DETECTIVE = re.compile(r"탐정님")
CALL_ASSISTANT = re.compile(r"조수님")

def rule_simulator(text, speaker):
    """Pure-regex approximation of ins-korean-naturalness LogicQA verdict."""
    if speaker == "detective":
        # Expect 하오체
        if CALL_DETECTIVE.search(text):
            return "FAIL", "호칭 누출 (탐정이 자기를 탐정님으로 지칭)"
        if HAEYOCHE_ENDINGS.search(text):
            return "FAIL", "해요체 혼입 (탐정 발화)"
        if BANMAL_ENDINGS.search(text) and not HAOCHE_ENDINGS.search(text):
            return "FAIL", "반말 혼입"
        if HAOCHE_ENDINGS.search(text):
            return "PASS", ""
        return "PASS", ""  # 모호 시 PASS
    elif speaker == "assistant":
        if CALL_ASSISTANT.search(text):
            return "FAIL", "호칭 누출 (조수가 자기를 조수님으로 지칭)"
        if BANMAL_ENDINGS.search(text) and not HAEYOCHE_ENDINGS.search(text):
            return "FAIL", "반말 혼입 (조수 발화)"
        if HAEYOCHE_ENDINGS.search(text):
            return "PASS", ""
        return "PASS", ""
    return "PASS", ""

@pytest.fixture(scope="module")
def samples():
    return json.loads(KOREAN_SAMPLES_PATH.read_text("utf-8"))

def test_negative_10_at_least_9_fail(samples):
    results = [rule_simulator(s["text"], s["speaker"]) for s in samples["negative"]]
    fail_count = sum(1 for v, _ in results if v == "FAIL")
    assert fail_count >= 9, f"Expected ≥9 FAIL on negative, got {fail_count}: {results}"

def test_positive_10_at_least_8_pass(samples):
    results = [rule_simulator(s["text"], s["speaker"]) for s in samples["positive"]]
    pass_count = sum(1 for v, _ in results if v == "PASS")
    assert pass_count >= 8, f"Expected ≥8 PASS on positive, got {pass_count}: {results}"
```

NOTE: Since rule_simulator is a regex-only stand-in for the actual LLM-backed ins-korean-naturalness, sample text in korean_speech_samples.json MUST be craftable such that regex alone achieves ≥9 FAIL on negatives + ≥8 PASS on positives. Plan 01 Task 2 writes the samples → this plan verifies roundtrip. If tests fail, Plan 01 sample text must be adjusted (or ins-korean-naturalness regex bank expanded).
