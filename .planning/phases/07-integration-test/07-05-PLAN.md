## CircuitBreaker public API (verified from circuit_breaker.py 2026-04-19)

```python
@dataclass
class CircuitBreaker:
    name: str
    max_failures: int = 3
    cooldown_seconds: int = 300
    state: CircuitState = CircuitState.CLOSED   # init=False
    failure_count: int = 0                      # init=False

    def call(self, fn: Callable, *args, **kwargs) -> Any:
        # CLOSED: runs fn; exception increments failure_count; >= max_failures → trip OPEN.
        # OPEN: if _cooldown_remaining() >= 0 → raise CircuitBreakerOpenError.
        #       else → transition to HALF_OPEN, run probe.
        # HALF_OPEN: probe call; success closes breaker; failure re-opens with fresh cooldown.
```

"""TEST-03 secondary: strict > 300s cooldown boundary + HALF_OPEN probe.

Exact boundary contract (circuit_breaker.py:131):
  remaining >= 0 → CircuitBreakerOpenError (raise).
  remaining < 0  → transition to HALF_OPEN and run probe.

Therefore:
  elapsed == 299.0 → remaining == 1.0 → OPEN (blocked)
  elapsed == 300.0 → remaining == 0.0 → OPEN (strict >) still blocked
  elapsed == 300.001 → remaining < 0 → HALF_OPEN probe admitted
"""
from __future__ import annotations

from unittest.mock import patch

import pytest

from scripts.orchestrator.circuit_breaker import (
    CircuitBreaker,
    CircuitBreakerOpenError,
    CircuitState,
)


def _boom():
    raise RuntimeError("fail")


def _trip_to_open(cb: CircuitBreaker, mt, at: float = 1000.0):
    mt.return_value = at
    for _ in range(cb.max_failures):
        with pytest.raises(RuntimeError):
            cb.call(_boom)
    assert cb.state is CircuitState.OPEN


def test_blocked_at_zero_seconds_elapsed():
    cb = CircuitBreaker(name="kling", max_failures=3, cooldown_seconds=300)
    with patch("scripts.orchestrator.circuit_breaker.time.monotonic") as mt:
        _trip_to_open(cb, mt, at=1000.0)
        mt.return_value = 1000.0
        with pytest.raises(CircuitBreakerOpenError) as exc:
            cb.call(lambda: "x")
        assert exc.value.cooldown_remaining > 299.0


def test_blocked_at_299_seconds_elapsed():
    cb = CircuitBreaker(name="kling", max_failures=3, cooldown_seconds=300)
    with patch("scripts.orchestrator.circuit_breaker.time.monotonic") as mt:
        _trip_to_open(cb, mt, at=1000.0)
        mt.return_value = 1000.0 + 299.0
        with pytest.raises(CircuitBreakerOpenError):
            cb.call(lambda: "x")


def test_still_blocked_at_exact_300_seconds_strict_gt():
    """circuit_breaker.py:131 strict `>`: at elapsed == cooldown, remaining == 0, still OPEN."""
    cb = CircuitBreaker(name="kling", max_failures=3, cooldown_seconds=300)
    with patch("scripts.orchestrator.circuit_breaker.time.monotonic") as mt:
        _trip_to_open(cb, mt, at=1000.0)
        mt.return_value = 1000.0 + 300.0
        with pytest.raises(CircuitBreakerOpenError) as exc:
            cb.call(lambda: "x")
        # remaining should be ~0, not negative.
        assert exc.value.cooldown_remaining is not None
        assert exc.value.cooldown_remaining >= 0.0


def test_admits_probe_strictly_after_300_seconds():
    """At elapsed > 300s, HALF_OPEN probe is admitted; success closes breaker."""
    cb = CircuitBreaker(name="kling", max_failures=3, cooldown_seconds=300)
    with patch("scripts.orchestrator.circuit_breaker.time.monotonic") as mt:
        _trip_to_open(cb, mt, at=1000.0)
        mt.return_value = 1000.0 + 300.001
        # Probe succeeds → breaker transitions to CLOSED.
        result = cb.call(lambda: "recovered")
        assert result == "recovered"
        assert cb.state is CircuitState.CLOSED


def test_failed_probe_in_half_open_reopens_with_fresh_cooldown():
    """Per circuit_breaker.py docstring: HALF_OPEN failure re-opens with fresh cooldown."""
    cb = CircuitBreaker(name="kling", max_failures=3, cooldown_seconds=300)
    with patch("scripts.orchestrator.circuit_breaker.time.monotonic") as mt:
        _trip_to_open(cb, mt, at=1000.0)
        mt.return_value = 1000.0 + 300.001
        with pytest.raises(RuntimeError):
            cb.call(_boom)
        assert cb.state is CircuitState.OPEN
        # Next call is again blocked by cooldown.
        mt.return_value = 1000.0 + 300.002
        with pytest.raises(CircuitBreakerOpenError):
            cb.call(lambda: "x")


def test_no_freezegun_dependency_used():
    """RESEARCH §Environment Availability: pytest-socket + freezegun NOT installed;
    stdlib unittest.mock.patch is the ONLY tool."""
    import sys
    assert "freezegun" not in sys.modules or "freezegun" not in dir()
