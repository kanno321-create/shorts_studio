"""TEST-01 SC1: E2E mock pipeline TREND → COMPLETE in one pass.

Structural peer of tests/phase05/test_pipeline_e2e_mock.py but uses Phase 7
mocks (Wave 1) to validate the mock-contract is compatible with the Phase 5
ShortsPipeline expectations. Proves:
  - dispatched_count == 13 (13 operational gates)
  - final_gate == "COMPLETE"
  - fallback_count == 0 (happy path)
  - 14 checkpoint files (13 + COMPLETE) on disk
  - failures.md never touched
  - Real network call = zero (MagicMock adapters + fake API keys)
"""
from __future__ import annotations

from pathlib import Path
from unittest.mock import MagicMock, patch

import pytest

from scripts.orchestrator import GateName, ShortsPipeline, Verdict
from tests.phase07.mocks.elevenlabs_mock import ElevenLabsMock
from tests.phase07.mocks.kling_mock import KlingMock
from tests.phase07.mocks.runway_mock import RunwayMock
from tests.phase07.mocks.shotstack_mock import ShotstackMock
from tests.phase07.mocks.typecast_mock import TypecastMock


def _pass_verdict() -> Verdict:
    return Verdict(
        result="PASS", score=90, evidence=[],
        semantic_feedback="", inspector_name="shorts-supervisor",
    )


def _producer_output(tmp_path: Path) -> dict:
    return {
        "artifact_path": str(tmp_path / "artifact.json"),
        "prompt": "dark detective scene",
        "duration_seconds": 5,
        "anchor_frame": str(tmp_path / "a.png"),
    }


def _build_pipeline(tmp_path: Path, session_id: str) -> ShortsPipeline:
    producer = MagicMock(return_value=_producer_output(tmp_path))
    supervisor = MagicMock(return_value=_pass_verdict())

    kling = KlingMock()
    runway = RunwayMock()
    typecast = TypecastMock()
    elevenlabs = ElevenLabsMock()
    shotstack = ShotstackMock()

    pipeline = ShortsPipeline(
        session_id=session_id,
        state_root=tmp_path / "state",
        failures_path=tmp_path / "failures.md",
        kling_adapter=kling,
        runway_adapter=runway,
        typecast_adapter=typecast,
        elevenlabs_adapter=elevenlabs,
        shotstack_adapter=shotstack,
        producer_invoker=producer,
        supervisor_invoker=supervisor,
        asset_sourcer_invoker=lambda prompt: tmp_path / "stock.png",
    )
    return pipeline


def test_full_pipeline_runs_13_gates_and_completes(
    tmp_path: Path, _fake_env: None, tmp_session_id: str,
):
    """TEST-01 primary assertion: TREND → COMPLETE in one mock pass."""
    pipeline = _build_pipeline(tmp_path, tmp_session_id)

    with patch.object(pipeline.timeline, "align", return_value=[]), \
         patch.object(pipeline.timeline, "insert_transition_shots", return_value=[]):
        result = pipeline.run()

    assert result["session_id"] == tmp_session_id
    assert result["final_gate"] == "COMPLETE", (
        f"Expected COMPLETE, got {result['final_gate']}; result={result}"
    )
    assert result["dispatched_count"] == 13, (
        f"Expected 13 operational gates dispatched (RESEARCH Correction 1), "
        f"got {result['dispatched_count']}"
    )
    assert result["fallback_count"] == 0, "Happy path must NOT insert any fallback"


def test_checkpoint_14_files_exist(
    tmp_path: Path, _fake_env: None, tmp_session_id: str,
):
    """D-6: 13 operational + 1 COMPLETE = 14 gate_NN.json files."""
    pipeline = _build_pipeline(tmp_path, tmp_session_id)

    with patch.object(pipeline.timeline, "align", return_value=[]), \
         patch.object(pipeline.timeline, "insert_transition_shots", return_value=[]):
        pipeline.run()

    session_dir = tmp_path / "state" / tmp_session_id
    assert session_dir.exists(), f"state dir missing: {session_dir}"
    gate_files = sorted(session_dir.glob("gate_*.json"))
    assert len(gate_files) == 14, (
        f"Expected 14 checkpoint files (13 operational + COMPLETE), got {len(gate_files)}: "
        f"{[f.name for f in gate_files]}"
    )


def test_failures_md_empty_on_happy_path(
    tmp_path: Path, _fake_env: None, tmp_session_id: str,
):
    """Happy path never appends to FAILURES.md."""
    pipeline = _build_pipeline(tmp_path, tmp_session_id)

    with patch.object(pipeline.timeline, "align", return_value=[]), \
         patch.object(pipeline.timeline, "insert_transition_shots", return_value=[]):
        pipeline.run()

    failures_path = tmp_path / "failures.md"
    # Either does not exist or is empty/only-frontmatter.
    if failures_path.exists():
        text = failures_path.read_text(encoding="utf-8")
        assert "FAIL after regeneration exhausted" not in text, (
            f"Happy path appended to failures.md:\n{text}"
        )


def test_checkpoint_files_round_trip_json(
    tmp_path: Path, _fake_env: None, tmp_session_id: str,
):
    """D-6: every gate_NN.json round-trips as JSON with required keys."""
    import json
    pipeline = _build_pipeline(tmp_path, tmp_session_id)

    with patch.object(pipeline.timeline, "align", return_value=[]), \
         patch.object(pipeline.timeline, "insert_transition_shots", return_value=[]):
        pipeline.run()

    session_dir = tmp_path / "state" / tmp_session_id
    for cp_file in sorted(session_dir.glob("gate_*.json")):
        data = json.loads(cp_file.read_text(encoding="utf-8"))
        required = {"_schema", "session_id", "gate", "gate_index", "timestamp", "verdict", "artifacts"}
        missing = required - set(data.keys())
        assert not missing, f"{cp_file.name} missing keys: {missing}"


def test_no_real_network_via_fake_env(
    tmp_path: Path, _fake_env: None, tmp_session_id: str,
):
    """D-1: Real network call = 0. API keys are fake; adapters are Phase 7 mocks."""
    import os
    for key in ("KLING_API_KEY", "RUNWAY_API_KEY", "TYPECAST_API_KEY",
                "ELEVENLABS_API_KEY", "SHOTSTACK_API_KEY"):
        assert os.environ.get(key) == "fake"

    pipeline = _build_pipeline(tmp_path, tmp_session_id)
    with patch.object(pipeline.timeline, "align", return_value=[]), \
         patch.object(pipeline.timeline, "insert_transition_shots", return_value=[]):
        result = pipeline.run()
    assert result["final_gate"] == "COMPLETE"


def test_korean_session_id_cp949_roundtrip(
    tmp_path: Path, _fake_env: None,
):
    """D-22: non-ASCII session id survives atomic write on Windows."""
    sid = "tst_phase07_한국어_e2e"
    pipeline = _build_pipeline(tmp_path, sid)
    with patch.object(pipeline.timeline, "align", return_value=[]), \
         patch.object(pipeline.timeline, "insert_transition_shots", return_value=[]):
        result = pipeline.run()
    assert result["session_id"] == sid
    # Session dir exists under the Korean name (file-system cp949 safety).
    assert (tmp_path / "state" / sid).exists()
