## NotebookLMFallbackChain contract (D-5 canonical)
```python
class NotebookLMFallbackChain:
    def __init__(self, backends: list[QueryBackend] | None = None): ...

    def query(self, question: str, notebook_id: str) -> tuple[str, int]:
        """Returns (answer, tier_used). Iterates backends in order.
        tier_used: 0=RAG, 1=grep, 2=hardcoded.
        Raises RuntimeError('all NotebookLM fallback tiers exhausted') only if every backend raised.
        """
```

## Failure modes (RESEARCH Area 4 lines 634-642)
- Tier 0 fails: subprocess rc!=0 / timeout / auth expired / notebook_id missing → try Tier 1
- Tier 1 fails: no markdown files match ALL keywords → try Tier 2
- Tier 2 never fails (returns sentinel for unknown ids)
- All 3 fail: only if backends list empty or all raise → RuntimeError

```python
class FakeFailing:
    def __init__(self, label: str): self.label = label

        raise RuntimeError(f"simulated fail: {self.label}")

def test_hardcoded_known_id_returns_seeded_answer():
    h = HardcodedDefaultsBackend()
    answer = h.query("아무거나", "naberal-shorts-channel-bible")

    assert "navy" in answer
    assert "cinematic" in answer


def test_hardcoded_unknown_id_returns_sentinel():
    h = HardcodedDefaultsBackend()
    answer = h.query("q", "unknown-notebook-xxx")
    assert answer == "fallback defaults unavailable for notebook_id=unknown-notebook-xxx"


def test_hardcoded_never_raises():
    h = HardcodedDefaultsBackend()

def test_grep_wiki_finds_keyword_intersection(tmp_wiki_dir: Path):
    (tmp_wiki_dir / "algorithm" / "note_a.md").write_text(
        "retention hook 완주율 signal 알고리즘", encoding="utf-8"
    )
    (tmp_wiki_dir / "kpi" / "note_b.md").write_text(
        "Different content entirely", encoding="utf-8"
    )
```

```python
from __future__ import annotations
import subprocess
from pathlib import Path

import pytest

from scripts.notebooklm.fallback import NotebookLMFallbackChain, HardcodedDefaultsBackend, RAGBackend, GrepWikiBackend


def test_real_rag_backend_falls_through_on_auth_failure(
    monkeypatch, tmp_path: Path, mock_notebooklm_skill_env: Path
):
    """D-5 acceptance: forced subprocess rc=1 -> chain activates tier>=1."""
    # Make the fake skill run.py fail with auth error
    run_py = mock_notebooklm_skill_env / "scripts" / "run.py"
    run_py.write_text(
        "import sys\nsys.stderr.write('not authenticated')\nsys.exit(1)\n",
        encoding="utf-8",
    )
    # Seed a tiny wiki dir so Tier 1 grep has something
    wiki_dir = tmp_path / "wiki"
    (wiki_dir / "continuity_bible").mkdir(parents=True)
    (wiki_dir / "continuity_bible" / "channel_identity.md").write_text(
        "---\ncategory: continuity_bible\nstatus: ready\ntags: [x]\nupdated: 2026-04-19\nsource_notebook: naberal-shorts-channel-bible\n---\n"
        "# Channel Identity\n색상 팔레트 navy + gold. 한국 시니어 타겟.",
        encoding="utf-8",
    )
    chain = NotebookLMFallbackChain(backends=[
        RAGBackend(),
        GrepWikiBackend(wiki_root=wiki_dir),
        HardcodedDefaultsBackend(),
    ])
    # Question has keywords that match the seeded wiki file
    answer, tier = chain.query("색상 팔레트 한국", "naberal-shorts-channel-bible")
    assert tier in (1, 2), f"expected fall-through (tier >=1), got tier={tier}"
    # Either tier 1 grep hit something, or tier 2 served hardcoded
    assert answer, "non-empty answer expected from fallback"


def test_real_rag_backend_falls_through_to_defaults_when_wiki_empty(
    monkeypatch, tmp_path: Path, mock_notebooklm_skill_env: Path
):
    """RAG fails + grep finds nothing -> tier 2 hardcoded returns known default."""
    (mock_notebooklm_skill_env / "scripts" / "run.py").write_text(
        "import sys\nsys.exit(1)\n", encoding="utf-8"
    )
    empty_wiki = tmp_path / "empty_wiki"
    empty_wiki.mkdir()
    chain = NotebookLMFallbackChain(backends=[
        RAGBackend(),
        GrepWikiBackend(wiki_root=empty_wiki),
        HardcodedDefaultsBackend(),
    ])
    answer, tier = chain.query("어떤 색 팔레트가 적절한가", "naberal-shorts-channel-bible")
    assert tier == 2
    assert "navy" in answer or "cinematic" in answer


def test_rag_success_returns_tier_0(
    monkeypatch, tmp_path: Path, mock_notebooklm_skill_env: Path
):
    """When subprocess rc=0, tier=0 short-circuits the chain."""
    (mock_notebooklm_skill_env / "scripts" / "run.py").write_text(
        "import sys\nsys.stdout.write('real RAG answer')\nsys.exit(0)\n",
        encoding="utf-8",
    )
    wiki_dir = tmp_path / "wiki"
    wiki_dir.mkdir()
    chain = NotebookLMFallbackChain(backends=[
        RAGBackend(),
        GrepWikiBackend(wiki_root=wiki_dir),
        HardcodedDefaultsBackend(),
    ])
    answer, tier = chain.query("test question", "naberal-shorts-channel-bible")
    assert tier == 0
    assert "real RAG answer" in answer
```
