"""Tests proving pre_tool_use.py Hook denies T2V patterns per VIDEO-01 / D-13.

Regression guard: if .claude/deprecated_patterns.json loses the T2V regex, these tests fail loudly.
"""
from __future__ import annotations
import json
import subprocess
import sys
import pytest
from pathlib import Path

HOOK = Path(__file__).resolve().parents[2] / ".claude" / "hooks" / "pre_tool_use.py"
STUDIO_ROOT = HOOK.parents[1].parent


def _hook_check(tool_name: str, content: str, file_path: str = "scripts/orchestrator/shorts_pipeline.py") -> dict:
    payload = {"tool_name": tool_name, "input": {"content": content, "file_path": file_path}}
    proc = subprocess.run(
        [sys.executable, str(HOOK)],
        input=json.dumps(payload),
        text=True, capture_output=True, timeout=5, cwd=str(STUDIO_ROOT),
    )
    assert proc.returncode == 0, f"Hook exited {proc.returncode}: {proc.stderr}"
    return json.loads(proc.stdout)


def test_t2v_word_denied():
    """D-13: bare 't2v' in source (e.g. function name) is blocked."""
    res = _hook_check("Write", "def make_t2v_clip(): pass")
    assert res["decision"] == "deny", f"VIDEO-01 regression: 't2v' allowed: {res}"


def test_text_to_video_denied():
    res = _hook_check("Write", "def text_to_video(prompt: str) -> str:\n    return ''")
    assert res["decision"] == "deny", f"VIDEO-01 regression: 'text_to_video' allowed: {res}"


def test_text2video_denied():
    res = _hook_check("Edit", "# Calling the text2video API endpoint")
    assert res["decision"] == "deny", f"VIDEO-01 regression: 'text2video' allowed: {res}"


def test_t2v_inside_comment_denied():
    """Hook regex matches anywhere — comment inclusion does NOT exempt."""
    res = _hook_check("Write", "# legacy note: we used to support T2V here")
    assert res["decision"] == "deny"


def test_t2v_inside_python_string_denied():
