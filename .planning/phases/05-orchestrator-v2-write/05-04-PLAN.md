```python
from __future__ import annotations
from dataclasses import dataclass, field, asdict
from typing import Literal
from pathlib import Path
from datetime import datetime, timezone

from .gates import (
    GateName,
    GATE_DEPS,
    GateFailure,
    MissingVerdict,
    GateDependencyUnsatisfied,
)
from .checkpointer import Checkpointer, Checkpoint, sha256_file


@dataclass
class Verdict:
    """Inspector output contract. Matches .claude/agents/_shared/rubric-schema.json draft-07.

    Fields:
      result: "PASS" or "FAIL" (enforced by type alias)
      score: 0-100 (not range-enforced here; rubric JSON schema does the bounds)
      evidence: list of violations (each a {rule, line?, detail, ...} dict)
      semantic_feedback: VQQA gradient narrative for Producer retry loop (RUB-03)
      inspector_name: originating Inspector id or "shorts-supervisor" for aggregated verdicts
    """
    result: Literal["PASS", "FAIL"]
    score: int
    evidence: list[dict]
    semantic_feedback: str
    inspector_name: str | None = None


# 13 operational gates — NOT IDLE, NOT COMPLETE. verify_all_dispatched requires all these.
_OPERATIONAL_GATES: frozenset[GateName] = frozenset(
    g for g in GateName if g not in (GateName.IDLE, GateName.COMPLETE)
)


class GateGuard:
    """Single enforcement point for GATE dispatch (ORCH-03, ORCH-04, ORCH-07 runtime).

    Pattern (per D-3):
      1. Inspector produces Verdict
      2. Pipeline calls GateGuard.dispatch(gate, verdict)
      3. dispatch() raises GateFailure on FAIL, or saves checkpoint + marks dispatched on PASS
      4. Pipeline calls ensure_dependencies() before each GATE
      5. Pipeline calls verify_all_dispatched() before transitioning to COMPLETE
    """

    def __init__(self, checkpointer: Checkpointer, session_id: str):
        self.cp = checkpointer
        self.session_id = session_id
        self._dispatched: set[GateName] = set()

    @property
    def dispatched(self) -> set[GateName]:
        """Read-only view of the dispatched set."""
        return set(self._dispatched)

    def dispatch(self, gate: GateName, verdict: Verdict | None, artifact_path: Path | None = None) -> None:
        """Single transition point — raises on FAIL, writes checkpoint on PASS.

        Raises:
          MissingVerdict — verdict is None (Inspector was not invoked)
          GateFailure    — verdict.result == "FAIL" (Inspector rejected)
        """
        if verdict is None:
            raise MissingVerdict(
                f"{gate.name}: Inspector must be called before dispatch (D-3)"
            )
        if verdict.result == "FAIL":
            raise GateFailure(gate, verdict.evidence)

        # PASS path — checkpoint + mark dispatched
        artifacts: dict = {}
        if artifact_path is not None and artifact_path.exists():
            artifacts = {
                "path": str(artifact_path).replace("\\", "/"),  # RESEARCH Pitfall 6: forward-slash for API
                "sha256": sha256_file(artifact_path),
            }

        cp = Checkpoint(
            session_id=self.session_id,
            gate=gate.name,
            gate_index=int(gate),
            timestamp=datetime.now(timezone.utc).isoformat(),
            verdict=asdict(verdict),
            artifacts=artifacts,
        )
        self.cp.save(cp)
        self._dispatched.add(gate)

    def verify_all_dispatched(self) -> bool:
        """COMPLETE gate precondition (ORCH-04, D-4).

        Returns True only when all 13 operational gates are dispatched.
        IDLE and COMPLETE are excluded by construction.
        """
        return _OPERATIONAL_GATES.issubset(self._dispatched)

    def ensure_dependencies(self, gate: GateName) -> None:
        """Runtime DAG enforcement (ORCH-07, D-7).

        Raises GateDependencyUnsatisfied if any declared prerequisite has not been dispatched.
        IDLE is treated as always-satisfied (it is the root).
        """
        missing = [
            d for d in GATE_DEPS[gate]
            if d not in self._dispatched and d is not GateName.IDLE
        ]
        if missing:
            raise GateDependencyUnsatisfied(
                f"{gate.name} needs: {[g.name for g in missing]}"
            )

    def missing_for_complete(self) -> set[GateName]:
        """Diagnostic helper — gates still needed before COMPLETE is reachable."""
        return set(_OPERATIONAL_GATES) - self._dispatched
```

- IDLE exclusion from `missing` check in `ensure_dependencies` — IDLE is the root, never dispatched but always "complete" as a prerequisite.

from __future__ import annotations
import pytest
from pathlib import Path
from scripts.orchestrator import GateName, GateFailure, MissingVerdict, GateDependencyUnsatisfied
from scripts.orchestrator.gate_guard import GateGuard, Verdict
from scripts.orchestrator.checkpointer import Checkpointer


def _pass_verdict(score: int = 92) -> Verdict:
    return Verdict(
        result="PASS",
        score=score,
        evidence=[],
        semantic_feedback="all clean",
        inspector_name="shorts-supervisor",
    )


def _fail_verdict(violations: int = 2) -> Verdict:
    return Verdict(
        result="FAIL",
        score=42,
        evidence=[{"rule": f"HC-{i}", "line": 10 + i, "detail": "problem"} for i in range(violations)],
        semantic_feedback="two structural failures",
        inspector_name="shorts-supervisor",
    )


def _make_guard(tmp_state_dir: Path, session_id: str = "s1") -> GateGuard:
    return GateGuard(Checkpointer(tmp_state_dir), session_id)


def test_dispatch_none_verdict_raises_missing(tmp_state_dir):
    g = _make_guard(tmp_state_dir)
    with pytest.raises(MissingVerdict) as ei:
        g.dispatch(GateName.TREND, None)
    assert "TREND" in str(ei.value)


def test_dispatch_fail_verdict_raises_gate_failure(tmp_state_dir):
    g = _make_guard(tmp_state_dir)
    v = _fail_verdict(violations=3)
    with pytest.raises(GateFailure) as ei:
        g.dispatch(GateName.TREND, v)
    assert ei.value.gate == GateName.TREND
    assert len(ei.value.evidence) == 3
    assert ei.value.evidence[0]["rule"] == "HC-0"


def test_dispatch_fail_does_not_checkpoint(tmp_state_dir):
    g = _make_guard(tmp_state_dir)
    with pytest.raises(GateFailure):
        g.dispatch(GateName.TREND, _fail_verdict())
    # No state file should exist
    session_dir = tmp_state_dir / "s1"
    assert not session_dir.exists() or not list(session_dir.glob("gate_*.json"))


def test_dispatch_fail_does_not_mark_dispatched(tmp_state_dir):
    g = _make_guard(tmp_state_dir)
    with pytest.raises(GateFailure):
        g.dispatch(GateName.TREND, _fail_verdict())
    assert GateName.TREND not in g.dispatched


def test_dispatch_pass_saves_checkpoint(tmp_state_dir):
    g = _make_guard(tmp_state_dir)
    g.dispatch(GateName.TREND, _pass_verdict())
    target = tmp_state_dir / "s1" / "gate_01.json"
    assert target.exists()


def test_dispatch_pass_marks_dispatched(tmp_state_dir):
    g = _make_guard(tmp_state_dir)
    g.dispatch(GateName.TREND, _pass_verdict())
    assert GateName.TREND in g.dispatched


def test_dispatch_artifact_sha256_recorded(tmp_state_dir, tmp_path):
    artifact = tmp_path / "asset.json"
    artifact.write_bytes(b"hello world")
    g = _make_guard(tmp_state_dir)
    g.dispatch(GateName.TREND, _pass_verdict(), artifact_path=artifact)
    import json
    data = json.loads((tmp_state_dir / "s1" / "gate_01.json").read_text(encoding="utf-8"))
    assert data["artifacts"]["sha256"] == "b94d27b9934d3e08a52e52d7da7dabfac484efe37a5380ee9088f7ace2efcde9"
    assert "/" in data["artifacts"]["path"] or "\\" not in data["artifacts"]["path"]

from __future__ import annotations
import pytest
from scripts.orchestrator import GateName
from scripts.orchestrator.gate_guard import GateGuard, Verdict
from scripts.orchestrator.checkpointer import Checkpointer


def _pass():
    return Verdict(result="PASS", score=90, evidence=[], semantic_feedback="", inspector_name="shorts-supervisor")


def test_empty_dispatch_is_false(tmp_state_dir):
    g = GateGuard(Checkpointer(tmp_state_dir), "s1")
    assert g.verify_all_dispatched() is False


def test_all_13_operational_dispatch_is_true(tmp_state_dir):
    g = GateGuard(Checkpointer(tmp_state_dir), "s1")
    operational = [gate for gate in GateName if gate not in (GateName.IDLE, GateName.COMPLETE)]
    assert len(operational) == 13
    for gate in operational:
        g.dispatch(gate, _pass())
    assert g.verify_all_dispatched() is True


def test_missing_one_gate_is_false(tmp_state_dir):
    g = GateGuard(Checkpointer(tmp_state_dir), "s1")
    operational = [gate for gate in GateName if gate not in (GateName.IDLE, GateName.COMPLETE)]
    for gate in operational[:-1]:  # all except MONITOR
        g.dispatch(gate, _pass())
    assert g.verify_all_dispatched() is False


def test_idle_dispatched_alone_is_not_sufficient(tmp_state_dir):
    g = GateGuard(Checkpointer(tmp_state_dir), "s1")
    # Try to dispatch IDLE (unusual but not forbidden by GateGuard)
    g.dispatch(GateName.IDLE, _pass())
    assert g.verify_all_dispatched() is False


def test_missing_for_complete_reports_gap(tmp_state_dir):
    g = GateGuard(Checkpointer(tmp_state_dir), "s1")
    g.dispatch(GateName.TREND, _pass())
    missing = g.missing_for_complete()
    assert GateName.NICHE in missing
    assert GateName.MONITOR in missing
    assert GateName.TREND not in missing
    assert GateName.IDLE not in missing
    assert GateName.COMPLETE not in missing


def test_complete_itself_not_required(tmp_state_dir):
    g = GateGuard(Checkpointer(tmp_state_dir), "s1")
    operational = [gate for gate in GateName if gate not in (GateName.IDLE, GateName.COMPLETE)]
    for gate in operational:
        g.dispatch(gate, _pass())
    # Do NOT dispatch COMPLETE
    assert g.verify_all_dispatched() is True
    assert GateName.COMPLETE not in g.missing_for_complete()


def test_operational_count_is_exactly_13():
    """Sanity check against RESEARCH §2 (13 operational excluding IDLE + COMPLETE)."""
    operational = [g for g in GateName if g not in (GateName.IDLE, GateName.COMPLETE)]
    assert len(operational) == 13
