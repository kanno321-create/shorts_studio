"""SC 2 + SC 5 + D-8/D-9/D-13 grep contract tests.

Uses subprocess grep (or pure-Python recursive read) to assert absent patterns
across scripts/orchestrator/ and scripts/hc_checks/.

If any of these fail, Phase 5 cannot ship.
"""
from __future__ import annotations
import re
import subprocess
import pytest
from pathlib import Path


REPO = Path(__file__).resolve().parents[2]
ORCH = REPO / "scripts" / "orchestrator"
HC = REPO / "scripts" / "hc_checks"


def _grep_recursive(pattern: str, *paths: Path, case_insensitive: bool = False) -> list[str]:
    """Pure-Python recursive grep. Returns list of matching 'file:line' strings."""
    flags = re.IGNORECASE if case_insensitive else 0
    regex = re.compile(pattern, flags)
    matches = []
    for root in paths:
        if not root.exists():
            continue
        for p in root.rglob("*.py"):
            try:
                text = p.read_text(encoding="utf-8", errors="replace")
            except Exception:
                continue
            for i, line in enumerate(text.splitlines(), 1):
                if regex.search(line):
                    matches.append(f"{p.relative_to(REPO)}:{i}: {line.strip()}")
    return matches


def test_no_skip_gates_in_orchestrator():
    """SC 2 / D-8: skip_gates must not appear in scripts/orchestrator/."""
    matches = _grep_recursive(r"skip_gates", ORCH)
    assert matches == [], f"SC 2 VIOLATED: skip_gates found:\n" + "\n".join(matches)


def test_no_t2v_in_orchestrator():
    """SC 5 / D-13: t2v / text_to_video / text2video must not appear in scripts/orchestrator/."""
    # Use case-insensitive search matching Hook regex semantics
    matches = _grep_recursive(r"\bt2v\b|text_to_video|text2video", ORCH, case_insensitive=True)
    assert matches == [], f"SC 5 VIOLATED: T2V patterns found:\n" + "\n".join(matches)


def test_no_todo_next_session_in_orchestrator():
    """D-9: TODO(next-session) must not appear."""

    assert matches == [], f"D-9 VIOLATED: TODO(next-session) found:\n" + "\n".join(matches)

def test_no_silent_except_pass_in_orchestrator():
    """Project Rule 3: no try-except Exception pass (silent fallback)."""
    # Looser check — catches 'except Exception:\n    pass' spanning 2 lines is hard with line-based grep.
    # Check for literal 'except Exception:' followed by 'pass' as weakest heuristic.
