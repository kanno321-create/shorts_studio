**Structure (RESEARCH §Architecture Pattern 3 + §4 Checkpointer Design):**
```python
from __future__ import annotations
import os
import json
import hashlib
from pathlib import Path
from datetime import datetime, timezone
from dataclasses import dataclass, asdict

SCHEMA_VERSION = 1

@dataclass
class Checkpoint:
    """ORCH-05 per-GATE persistence record (D-5).

    Schema:
      session_id: opaque string (D-5 path namespace)
      gate: GateName.name string (e.g. "NICHE")
      gate_index: GateName.value int (0-14)
      timestamp: ISO 8601 UTC with microseconds
      verdict: rubric-schema.json shape {result, score, evidence, semantic_feedback, inspector_name}
      artifacts: {"path": str, "sha256": str} or {} if no artifact
    """
    session_id: str
    gate: str
    gate_index: int
    timestamp: str
    verdict: dict
    artifacts: dict


class Checkpointer:
    """Atomic JSON-per-GATE persistence for Phase 5 orchestrator (ORCH-05).

    Path layout: {state_root}/{session_id}/gate_{gate_index:02d}.json
    Write: serialize to .tmp → os.replace() (atomic on Windows + POSIX).
    Resume: highest gate_index integer in the session dir (or -1).
    """
    SCHEMA_VERSION: int = 1

    def __init__(self, state_root: Path = Path("state")):
        self.root = Path(state_root)

    def save(self, cp: Checkpoint) -> Path:
        """Write one gate checkpoint atomically. Overwrites same (session, gate_index)."""
        target_dir = self.root / cp.session_id
        target_dir.mkdir(parents=True, exist_ok=True)
        target = target_dir / f"gate_{cp.gate_index:02d}.json"
        tmp = target.with_suffix(target.suffix + ".tmp")
        payload = {"_schema": self.SCHEMA_VERSION, **asdict(cp)}
        tmp.write_text(
            json.dumps(payload, ensure_ascii=False, indent=2),
            encoding="utf-8",
        )
        os.replace(tmp, target)  # atomic on Windows + POSIX (Python 3.3+ guarantee)
        return target

    def resume(self, session_id: str) -> int:
        """Highest gate_index in session dir, or -1 if session never created."""
        d = self.root / session_id
        if not d.exists():
            return -1
        indices: list[int] = []
        for p in d.glob("gate_*.json"):
            try:
                indices.append(int(p.stem.split("_")[1]))
            except (IndexError, ValueError):
                continue
        return max(indices) if indices else -1

    def dispatched_gates(self, session_id: str) -> set[str]:
        """Set of GateName.name strings recorded in session dir."""
        d = self.root / session_id
        if not d.exists():
            return set()
        out: set[str] = set()
        for p in sorted(d.glob("gate_*.json")):
            try:
                data = json.loads(p.read_text(encoding="utf-8"))
                out.add(data["gate"])
            except (json.JSONDecodeError, KeyError):
                continue
        return out

    def load(self, session_id: str, gate_index: int) -> dict | None:
        """Read one gate checkpoint back. Returns None if file absent."""
        target = self.root / session_id / f"gate_{gate_index:02d}.json"
        if not target.exists():
            return None
        return json.loads(target.read_text(encoding="utf-8"))


def sha256_file(path: Path, chunk_size: int = 65536) -> str:
    """Streaming SHA256 hex digest for arbitrarily large files.

    Source pattern: .preserved/harvested/hc_checks_raw/hc_checks.py lines 438-450.
    """
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(chunk_size), b""):
            h.update(chunk)
    return h.hexdigest()


def make_timestamp() -> str:
    """ISO 8601 UTC with microseconds — D-5 timestamp format."""
    return datetime.now(timezone.utc).isoformat()
```

**Critical:**
- `os.replace()` NOT `shutil.move()` — latter is not atomic on Windows (RESEARCH line 664-665).
- `gate_{gate_index:02d}.json` — zero-padded to 2 digits for natural sort (RESEARCH §4 line 676, Plan 04 resume iteration depends on sort order).
- `json.dumps(..., ensure_ascii=False, indent=2)` — Korean characters in evidence must survive round-trip.
- `from datetime import datetime, timezone; datetime.now(timezone.utc).isoformat()` — tz-aware timestamp.

**Do NOT:**
- Do NOT use `shutil.move` — RESEARCH §4 explicitly forbids.
- Do NOT use `json.dumps(..., ensure_ascii=True)` — Korean characters would become \uXXXX escapes breaking grep-based debugging.
- Do NOT re-export Checkpointer from `scripts/orchestrator/__init__.py` yet — Plan 07 will do that. Keep `__init__.py` minimal (just gates exports per Plan 01).

from __future__ import annotations
import json
import os
import pytest
from pathlib import Path
from scripts.orchestrator.checkpointer import Checkpointer, Checkpoint, sha256_file


def _make_cp(gate_name: str = "TREND", gate_index: int = 1) -> Checkpoint:
    return Checkpoint(
        session_id="20260419-test",
        gate=gate_name,
        gate_index=gate_index,
        timestamp="2026-04-19T14:35:40.123456+00:00",
        verdict={"result": "PASS", "score": 88, "evidence": [], "semantic_feedback": "", "inspector_name": "shorts-supervisor"},
        artifacts={"path": "state/20260419-test/trend.json", "sha256": "a" * 64},

def test_save_creates_file(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    target = c.save(_make_cp())
    assert target.exists()
    assert target.name == "gate_01.json"  # zero-padded


def test_no_tmp_file_left_behind(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    c.save(_make_cp())
    tmp_files = list((tmp_state_dir / "20260419-test").glob("*.tmp"))
    assert tmp_files == [], f"os.replace() did not clean up: {tmp_files}"


def test_schema_version_embedded(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    target = c.save(_make_cp())
    data = json.loads(target.read_text(encoding="utf-8"))
    assert data["_schema"] == 1


def test_round_trip_preserves_all_fields(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    original = _make_cp()
    target = c.save(original)
    loaded = json.loads(target.read_text(encoding="utf-8"))
    assert loaded["session_id"] == original.session_id
    assert loaded["gate"] == original.gate
    assert loaded["gate_index"] == original.gate_index
    assert loaded["timestamp"] == original.timestamp
    assert loaded["verdict"] == original.verdict
    assert loaded["artifacts"] == original.artifacts


def test_korean_evidence_preserved(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    cp = _make_cp()
    cp.verdict["semantic_feedback"] = "탐정님의 대사가 해요체로 이탈하고 있습니다"
    target = c.save(cp)
    raw = target.read_text(encoding="utf-8")
    assert "탐정님" in raw, "Korean characters must not be escaped to \\uXXXX"
    loaded = json.loads(raw)
    assert loaded["verdict"]["semantic_feedback"] == cp.verdict["semantic_feedback"]


def test_zero_padded_filenames(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    # gate_index 3 → "gate_03.json" not "gate_3.json"
    target = c.save(_make_cp(gate_index=3))
    assert target.name == "gate_03.json"
    target = c.save(_make_cp(gate_index=14))
    assert target.name == "gate_14.json"


def test_idempotent_overwrite(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    cp = _make_cp()
    c.save(cp)
    cp.verdict["score"] = 99
    c.save(cp)  # same session_id + gate_index → overwrites
    files = list((tmp_state_dir / "20260419-test").glob("gate_01.json"))
    assert len(files) == 1
    loaded = json.loads(files[0].read_text(encoding="utf-8"))
    assert loaded["verdict"]["score"] == 99  # latest write wins


def test_sha256_file_stable(tmp_path):
    target = tmp_path / "probe.bin"
    target.write_bytes(b"hello world")
    h1 = sha256_file(target)
    h2 = sha256_file(target)
    assert h1 == h2
    assert len(h1) == 64  # hex digest length
    # Known SHA256 for "hello world" = b94d27b9934d3e08a52e52d7da7dabfac484efe37a5380ee9088f7ace2efcde9
    assert h1 == "b94d27b9934d3e08a52e52d7da7dabfac484efe37a5380ee9088f7ace2efcde9"


def test_sha256_streaming_large_file(tmp_path):

    target = tmp_path / "big.bin"
    target.write_bytes(b"A" * 200_000)  # 3+ chunks
    digest = sha256_file(target)
    assert len(digest) == 64

from __future__ import annotations
from scripts.orchestrator.checkpointer import Checkpointer, Checkpoint


def _make_cp(gate: str, idx: int) -> Checkpoint:
    return Checkpoint(
        session_id="s1",
        gate=gate,
        gate_index=idx,
        timestamp=f"2026-04-19T{idx:02d}:00:00+00:00",
        verdict={"result": "PASS", "score": 90, "evidence": [], "semantic_feedback": "", "inspector_name": "shorts-supervisor"},
        artifacts={},
    )


def test_resume_returns_minus_one_for_nonexistent_session(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    assert c.resume("never-existed") == -1


def test_resume_returns_minus_one_for_empty_directory(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    (tmp_state_dir / "s1").mkdir(parents=True)
    assert c.resume("s1") == -1


def test_resume_returns_highest_gate_index(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    c.save(_make_cp("IDLE", 0))
    c.save(_make_cp("TREND", 1))
    c.save(_make_cp("NICHE", 2))
    c.save(_make_cp("ASSETS", 8))
    assert c.resume("s1") == 8


def test_resume_not_confused_by_non_gate_files(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    c.save(_make_cp("TREND", 1))
    # Drop a junk file that is not a gate file
    (tmp_state_dir / "s1" / "debug.log").write_text("noise")
    (tmp_state_dir / "s1" / "gate_NOT_INT.json").write_text("{}")
    assert c.resume("s1") == 1


def test_dispatched_gates_empty(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    assert c.dispatched_gates("nobody") == set()


def test_dispatched_gates_collects_names(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    c.save(_make_cp("IDLE", 0))
    c.save(_make_cp("TREND", 1))
    c.save(_make_cp("NICHE", 2))
    assert c.dispatched_gates("s1") == {"IDLE", "TREND", "NICHE"}


def test_load_returns_none_for_missing_gate(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    c.save(_make_cp("TREND", 1))
    assert c.load("s1", 99) is None
    loaded = c.load("s1", 1)
    assert loaded is not None
    assert loaded["gate"] == "TREND"
    assert loaded["_schema"] == 1


def test_sessions_isolated(tmp_state_dir):
    c = Checkpointer(tmp_state_dir)
    cp1 = _make_cp("TREND", 1)
    cp2 = _make_cp("NICHE", 2)
    cp2.session_id = "s2"
    c.save(cp1)
    c.save(cp2)
    assert c.resume("s1") == 1
    assert c.resume("s2") == 2
    assert c.dispatched_gates("s1") == {"TREND"}
    assert c.dispatched_gates("s2") == {"NICHE"}
