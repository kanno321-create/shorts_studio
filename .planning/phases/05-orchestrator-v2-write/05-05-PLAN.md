## VoiceFirstTimeline API

```
class VoiceFirstTimeline:
    TRANSITION_TEMPLATES = ("prop_closeup", "silhouette", "background_wipe")
    MIN_CLIP_DURATION = 4.0
    MAX_CLIP_DURATION = 8.0
    MIN_SPEED = 0.8
    MAX_SPEED = 1.25
    MIN_TRANSITION_DURATION = 0.5
    MAX_TRANSITION_DURATION = 1.0
    TRANSITION_EVERY_N_CUTS = 3

    def __init__(self, rng_seed: int | None = None)
    def align(audio_segments, video_cuts) -> list[TimelineEntry]
    def insert_transition_shots(timeline) -> list[TimelineEntry | TransitionEntry]
    def integrated_render(*args, **kwargs)  # raises IntegratedRenderForbidden
```

6. `align(audio_segments, video_cuts) -> list[TimelineEntry]`:
   - If len mismatch: `raise TimelineMismatch(f"audio={len(audio_segments)} video={len(video_cuts)} ...")`
   - For each (aud, vid) pair in zip:
     - If `aud.duration` not in [MIN_CLIP_DURATION, MAX_CLIP_DURATION]: `raise InvalidClipDuration(f"audio segment {aud.index} is {aud.duration:.2f}s (must be in [4, 8])")`
     - If `vid.duration <= 0`: `raise ClipDurationMismatch(f"video cut {vid.index}: duration {vid.duration} invalid")`
     - Compute `speed = aud.duration / vid.duration`
     - If speed not in [MIN_SPEED, MAX_SPEED]: `raise ClipDurationMismatch(f"segment {aud.index}: speed adjust {speed:.2f} outside [0.8, 1.25] - regenerate clip")`
     - Append `TimelineEntry(start=aud.start, end=aud.end, clip_path=vid.path, speed=speed, audio_path=aud.path)`
   - Return list

7. `insert_transition_shots(timeline) -> list[TimelineItem]`:
   - Iterate with enumerate:
     - Append entry
     - `is_last = i >= len(timeline) - 1`
     - `should_insert = (i + 1) % TRANSITION_EVERY_N_CUTS == 0`
     - If `should_insert and not is_last`:
       - `template = self._rng.choice(TRANSITION_TEMPLATES)`
       - `duration = self._rng.uniform(MIN_TRANSITION_DURATION, MAX_TRANSITION_DURATION)`
       - Append `TransitionEntry(template=template, duration=duration, after_index=i)`
   - Return out

8. `integrated_render(self, *args, **kwargs)`:
   - `raise IntegratedRenderForbidden("D-10: audio+video simultaneous generation is forbidden. Use: align(audio_segments, video_cuts) -> insert_transition_shots(timeline) -> Shotstack.render()")`

Forbidden:
- No imports of pydub, httpx, requests, or any HTTP library (pure data transformation)
- No render() or render_all() method in this class (Plan 06 Shotstack handles that)
- No audio_from_video or video_from_audio helpers (D-10 forbids)

```python
from __future__ import annotations
from pathlib import Path
from scripts.orchestrator.voice_first_timeline import (
    VoiceFirstTimeline, TimelineEntry, TransitionEntry
)


def _entry(i: int) -> TimelineEntry:
    return TimelineEntry(
        start=i * 5.0,
        end=(i + 1) * 5.0,
        clip_path=Path(f"cut_{i}.mp4"),
        speed=1.0,
        audio_path=Path(f"aud_{i}.mp3"),
    )


def test_empty_timeline_returns_empty():
    t = VoiceFirstTimeline()
    assert t.insert_transition_shots([]) == []


def test_two_cuts_no_transitions():
    t = VoiceFirstTimeline(rng_seed=42)
    out = t.insert_transition_shots([_entry(0), _entry(1)])
    assert len(out) == 2
    assert all(isinstance(x, TimelineEntry) for x in out)


def test_three_cuts_no_transitions_because_third_is_last():
    t = VoiceFirstTimeline(rng_seed=42)
    # i=2 (3rd cut) - (i+1)%3==0 is True, but is_last=True - no transition
    out = t.insert_transition_shots([_entry(0), _entry(1), _entry(2)])
    transitions = [x for x in out if isinstance(x, TransitionEntry)]
    assert len(transitions) == 0


def test_four_cuts_one_transition_after_index_2():
    t = VoiceFirstTimeline(rng_seed=42)
    # i=2 (3rd cut): (i+1)%3==0, is_last=False -> insert transition after index 2
    out = t.insert_transition_shots([_entry(i) for i in range(4)])
    transitions = [x for x in out if isinstance(x, TransitionEntry)]
    assert len(transitions) == 1
    assert transitions[0].after_index == 2


def test_six_cuts_one_transition():
    t = VoiceFirstTimeline(rng_seed=42)
    # i=2: insert. i=5: (i+1)%3==0 but is_last=True -> no insert
    out = t.insert_transition_shots([_entry(i) for i in range(6)])
    transitions = [x for x in out if isinstance(x, TransitionEntry)]
    assert len(transitions) == 1


def test_seven_cuts_two_transitions():
    t = VoiceFirstTimeline(rng_seed=42)
    # i=2: insert. i=5: insert (because is_last False, cut 6 still ahead). i=6: (i+1)%3!=0.
    out = t.insert_transition_shots([_entry(i) for i in range(7)])
    transitions = [x for x in out if isinstance(x, TransitionEntry)]
    assert len(transitions) == 2
    assert [t.after_index for t in transitions] == [2, 5]


def test_transition_templates_only_from_three():
    t = VoiceFirstTimeline(rng_seed=42)
    out = t.insert_transition_shots([_entry(i) for i in range(10)])
    allowed = {"prop_closeup", "silhouette", "background_wipe"}
    for x in out:
        if isinstance(x, TransitionEntry):
            assert x.template in allowed


def test_transition_duration_in_range():
    t = VoiceFirstTimeline(rng_seed=42)
    out = t.insert_transition_shots([_entry(i) for i in range(10)])
    for x in out:

def test_determinism_with_same_seed():
    t1 = VoiceFirstTimeline(rng_seed=42)
    t2 = VoiceFirstTimeline(rng_seed=42)
    out1 = t1.insert_transition_shots([_entry(i) for i in range(10)])
    out2 = t2.insert_transition_shots([_entry(i) for i in range(10)])
    templates1 = [x.template for x in out1 if isinstance(x, TransitionEntry)]
    templates2 = [x.template for x in out2 if isinstance(x, TransitionEntry)]
    assert templates1 == templates2


def test_different_seeds_can_differ():
```
