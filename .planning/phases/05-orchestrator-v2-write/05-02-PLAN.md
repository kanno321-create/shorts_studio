```python
from __future__ import annotations
import time
from enum import Enum
from dataclasses import dataclass, asdict, field
from typing import Callable, TypeVar

from .gates import CircuitOpen  # import from Plan 01 module

T = TypeVar("T")

class CircuitState(str, Enum):
    CLOSED = "CLOSED"
    OPEN = "OPEN"
    HALF_OPEN = "HALF_OPEN"

@dataclass
class CircuitBreaker:
    """3-state circuit breaker (ORCH-06 / D-6).

    CLOSED: calls pass through, failures increment consecutive counter.
    OPEN: all calls raise CircuitOpen until cooldown_seconds have elapsed.
    HALF_OPEN: next call is a probe; success returns state to CLOSED,
               failure returns state to OPEN with fresh opened_at.

    Independent of regeneration loop (D-12). Wraps external API calls only.
    """
    name: str
    max_failures: int = 3            # D-6 exact
    cooldown_seconds: int = 300      # D-6 exact (5 min)
    state: CircuitState = CircuitState.CLOSED
    consecutive_failures: int = 0
    opened_at: float | None = None

    def call(self, fn: Callable[[], T]) -> T:
        now = time.time()
        if self.state is CircuitState.OPEN:
            assert self.opened_at is not None
            elapsed = now - self.opened_at
            if elapsed >= self.cooldown_seconds:
                # cooldown expired — probe
                self.state = CircuitState.HALF_OPEN
            else:
                remaining = self.cooldown_seconds - elapsed
                raise CircuitOpen(
                    f"CircuitBreaker '{self.name}' OPEN for {remaining:.0f}s more"
                )
        try:
            result = fn()
        except Exception:
            # failure path
            self.consecutive_failures += 1
            if self.state is CircuitState.HALF_OPEN or self.consecutive_failures >= self.max_failures:
                self.state = CircuitState.OPEN
                self.opened_at = time.time()
            raise
        # success path
        self.state = CircuitState.CLOSED
        self.consecutive_failures = 0
        self.opened_at = None
        return result

    def to_dict(self) -> dict:
        """Serialize for Checkpointer JSON storage (ORCH-05 integration, Plan 03)."""
        d = asdict(self)
        d["state"] = self.state.value
        return d

    @classmethod
    def from_dict(cls, d: dict) -> "CircuitBreaker":
        """Deserialize from Checkpointer JSON (cooldown timer survives process restart)."""
        d = {**d, "state": CircuitState(d["state"])}
        return cls(**d)

    def is_open(self) -> bool:
        """Convenience query (does NOT transition state)."""
        return self.state is CircuitState.OPEN

    def reset(self) -> None:
        """Admin reset — used by tests and manual recovery."""
        self.state = CircuitState.CLOSED
        self.consecutive_failures = 0
        self.opened_at = None
```

from __future__ import annotations
import pytest
from scripts.orchestrator.circuit_breaker import CircuitBreaker, CircuitState
from scripts.orchestrator import CircuitOpen


def _failing_call():
    raise RuntimeError("simulated API error")


def _successful_call():
    return "ok"


def test_initial_state_closed():
    b = CircuitBreaker("test")
    assert b.state is CircuitState.CLOSED
    assert b.consecutive_failures == 0
    assert b.opened_at is None


def test_default_constants_match_d6():
    b = CircuitBreaker("test")
    assert b.max_failures == 3, "D-6 locks max_failures=3"
    assert b.cooldown_seconds == 300, "D-6 locks cooldown=300s (5 min)"


def test_single_failure_stays_closed():
    b = CircuitBreaker("test")
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    assert b.state is CircuitState.CLOSED
    assert b.consecutive_failures == 1


def test_three_failures_opens():
    b = CircuitBreaker("test")
    for _ in range(3):
        with pytest.raises(RuntimeError):
            b.call(_failing_call)
    assert b.state is CircuitState.OPEN
    assert b.consecutive_failures == 3
    assert b.opened_at is not None


def test_open_raises_circuit_open_without_invoking_fn():
    b = CircuitBreaker("test", max_failures=1)
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    assert b.state is CircuitState.OPEN
    # Now OPEN: further calls should raise CircuitOpen BEFORE invoking fn
    calls = []
    def _tracker():
        calls.append(1)
        return "should not reach"
    with pytest.raises(CircuitOpen):
        b.call(_tracker)
    assert calls == [], "fn should NOT be invoked while OPEN"


def test_success_resets_failure_counter():
    b = CircuitBreaker("test")
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    assert b.consecutive_failures == 2
    # Now success
    result = b.call(_successful_call)
    assert result == "ok"
    assert b.consecutive_failures == 0
    assert b.state is CircuitState.CLOSED


def test_half_open_success_returns_to_closed(monkeypatch):
    b = CircuitBreaker("test", max_failures=1, cooldown_seconds=1)
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    assert b.state is CircuitState.OPEN
    # Simulate cooldown elapsed
    import time as _time
    monkeypatch.setattr(_time, "time", lambda: b.opened_at + 2.0)
    result = b.call(_successful_call)
    assert result == "ok"
    assert b.state is CircuitState.CLOSED
    assert b.consecutive_failures == 0


def test_half_open_failure_returns_to_open(monkeypatch):
    b = CircuitBreaker("test", max_failures=1, cooldown_seconds=1)
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    original_opened = b.opened_at
    import time as _time
    monkeypatch.setattr(_time, "time", lambda: original_opened + 2.0)
    # Probe fails
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    assert b.state is CircuitState.OPEN
    assert b.opened_at is not None
    assert b.opened_at != original_opened, "opened_at must refresh for new cooldown"


def test_to_dict_from_dict_round_trip():
    b = CircuitBreaker("kling", max_failures=3, cooldown_seconds=300)
    with pytest.raises(RuntimeError):
        b.call(_failing_call)
    d = b.to_dict()
    # JSON-safe: state is a string, not an enum
    assert d["state"] == "CLOSED"
    assert d["name"] == "kling"
    assert d["consecutive_failures"] == 1
    b2 = CircuitBreaker.from_dict(d)
    assert b2.state is CircuitState.CLOSED
    assert b2.name == "kling"
    assert b2.consecutive_failures == 1
    assert b2.max_failures == 3


def test_from_dict_preserves_open_state_with_opened_at():
    payload = {
        "name": "runway",
        "max_failures": 3,
        "cooldown_seconds": 300,
        "state": "OPEN",
        "consecutive_failures": 3,
        "opened_at": 1700000000.0,
    }
    b = CircuitBreaker.from_dict(payload)
    assert b.state is CircuitState.OPEN
    assert b.opened_at == 1700000000.0

from __future__ import annotations
import pytest
from scripts.orchestrator.circuit_breaker import CircuitBreaker, CircuitState
from scripts.orchestrator import CircuitOpen


def test_cooldown_not_elapsed_still_raises(monkeypatch):
    """OPEN with opened_at only 100s ago, cooldown=300s, call must still raise CircuitOpen."""
    b = CircuitBreaker("test", max_failures=1, cooldown_seconds=300)
    import time as _time
    fixed_now = 1_700_000_000.0
    monkeypatch.setattr(_time, "time", lambda: fixed_now)
    with pytest.raises(RuntimeError):
        b.call(lambda: (_ for _ in ()).throw(RuntimeError("x")))
    assert b.state is CircuitState.OPEN
    assert b.opened_at == fixed_now
    # 100s later — still inside cooldown
    monkeypatch.setattr(_time, "time", lambda: fixed_now + 100.0)
    with pytest.raises(CircuitOpen) as ei:
        b.call(lambda: "should not run")
    assert "more" in str(ei.value).lower() or "open" in str(ei.value).lower()


def test_cooldown_exactly_elapsed_probes_half_open(monkeypatch):
    """At exactly cooldown_seconds elapsed, the next call transitions to HALF_OPEN."""
    b = CircuitBreaker("test", max_failures=1, cooldown_seconds=300)
    import time as _time
    fixed_now = 1_700_000_000.0
    monkeypatch.setattr(_time, "time", lambda: fixed_now)
    with pytest.raises(RuntimeError):
        b.call(lambda: (_ for _ in ()).throw(RuntimeError("x")))
    # exactly cooldown elapsed
    monkeypatch.setattr(_time, "time", lambda: fixed_now + 300.0)
    result = b.call(lambda: "probe-success")
    assert result == "probe-success"
    assert b.state is CircuitState.CLOSED


def test_serialized_cooldown_survives_restart(monkeypatch):
    """Simulate process restart: serialize OPEN breaker, deserialize, verify cooldown still enforced."""
    b = CircuitBreaker("test", max_failures=1, cooldown_seconds=300)
    import time as _time
    fixed_now = 1_700_000_000.0
    monkeypatch.setattr(_time, "time", lambda: fixed_now)
    with pytest.raises(RuntimeError):
        b.call(lambda: (_ for _ in ()).throw(RuntimeError("x")))
    d = b.to_dict()
    # "restart" — rebuild from dict
    b2 = CircuitBreaker.from_dict(d)
    assert b2.state is CircuitState.OPEN
    # 100s later
    monkeypatch.setattr(_time, "time", lambda: fixed_now + 100.0)
    with pytest.raises(CircuitOpen):
        b2.call(lambda: "blocked")
