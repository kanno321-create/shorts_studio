Target VisualSpec schema (Pydantic v2):
```python
class ClipDesign(BaseModel):
    model_config = ConfigDict(extra='forbid')
    type: Literal["video", "image"]
    src: str                                   # "<episode>/filename.ext" — relative to remotion/public/<job_id>/
    durationInFrames: PositiveInt              # int only, float 금지 (Pitfall 6)
    transition: Literal["fade", "glitch", "rgb-split", "zoom-blur", "light-leak", "clock-wipe", "pixelate", "checkerboard"]
    movement: Literal["zoom_in", "zoom_out", "pan_left", "pan_right"] | None = None  # None = freeze (Pitfall 4)


class TitleKeyword(BaseModel):
    model_config = ConfigDict(extra='forbid')
    text: str
    color: str  # "#RRGGBB"


class VisualSpec(BaseModel):
    model_config = ConfigDict(extra='forbid')
    titleLine1: str                            # "50년 미제"
    titleLine2: str                            # "암호 살인마"
    titleKeywords: list[TitleKeyword]
    accentColor: str                           # "#FF2200"
    channelName: str                           # "사건기록부"
    hashtags: str                              # "#쇼츠 #범죄 #..."
    fontFamily: Literal["BlackHanSans", "DoHyeon", "NotoSansKR"]  # Google Fonts only (Pretendard 금지)
    characterLeftSrc: str                      # "zodiac-killer/character_assistant.png" (assistant — Q4 좌=조수)
    characterRightSrc: str                     # "zodiac-killer/character_detective.png" (detective — Q4 우=탐정)
    subtitlePosition: confloat(ge=0.0, le=1.0) # 0.8 (incidents 하단)
    subtitleHighlightColor: str                # "#FFFFFF"
    subtitleFontSize: PositiveInt              # 68
    audioSrc: str                              # "zodiac-killer/narration.mp3"
    durationInFrames: PositiveInt              # audio_duration * 30 (int)
    transitionType: Literal["fade", "glitch", "rgb-split", "zoom-blur", "light-leak", "clock-wipe", "pixelate", "checkerboard"]
    clips: list[ClipDesign]                    # zodiac-killer 기준 16 entries


class SourcesManifest(BaseModel):
    model_config = ConfigDict(extra='forbid')
    character_detective: str                   # "output/<episode>/sources/character_detective.png"
    character_assistant: str                   # "output/<episode>/sources/character_assistant.png"
    intro_signature: str                       # "output/<episode>/sources/intro_signature.mp4"
    outro_signature: str | None                # None = OutroCard.tsx 프로그램적 (Plan 16-03 Task 0 Option A)
    scene_sources: list[str]                   # ≥5 b-roll paths
    scene_sources_count: PositiveInt           # Pitfall 5: >= 5
    real_image_count: int                      # real photos
    veo_supplement_count: int                  # ≤ 2 (feedback_veo_supplementary_only)
    signature_reuse_count: int                 # intro + optional outro
    real_ratio: confloat(ge=0.0, le=1.0)       # real_image / (real + signature + veo)  — ≥ 0.75
```

    ```python
    def load_channel_preset(memory_path: Path) -> dict:
        """`.claude/memory/project_channel_preset_{channel}.md` 파싱하여 preset dict 반환.

        간이 파서 — frontmatter 이후 ##/### 헤딩 + 값 라인을 key-value 로 추출.
        복잡한 구조는 향후 YAML 변환 고려 (Phase 17+).

            "fontFamily": "BlackHanSans",

            "subtitleHighlightColor": "#FFFFFF",

            "transitions": VALID_TRANSITIONS,

        for key_re, field_key in (

            (r"^## accentColor.*?\n\s*`([^`]+)`", "accentColor"),
            (r"^## subtitlePosition\s*\n\s*`([0-9.]+)`", "subtitlePosition"),
            (r"^## subtitleHighlightColor\s*\n\s*`(#[0-9A-Fa-f]{6})`", "subtitleHighlightColor"),

            if m:

                if field_key in ("subtitlePosition",):
                    preset[field_key] = float(val)
                elif field_key in ("subtitleFontSize",):
                    preset[field_key] = int(val)
                else:

    def _build_clips(

            ))

            ))
    ```

    ```python
    """verify_baseline_parity — ffprobe 기반 우리 final.mp4 의 production baseline 정량 비교.

    shorts_naberal/output/{zodiac-killer, mary-celeste, db-cooper, elisa-lam, kitakyushu-matsunaga}/final.mp4
    를 baseline 으로 잡고 우리 산출 final.mp4 의 resolution / duration / bitrate / codec / 자막 coverage 를 비교.

    gate_guard.ASSEMBLY 최종 체크 또는 CLI 로 실행.

    Phase 16-04 — REQ-PROD-INT-13.

    Exit code:
      0 = PASS (모든 criterion 내 ±10%)
      1 = FAIL (1개 이상 criterion 미달)
      2 = baseline 접근 불가 (shorts_naberal 경로 없음 — skip-like)
    """
    from __future__ import annotations

    import argparse
    import json
    import logging
    import shutil
    import subprocess
    import sys
    from dataclasses import dataclass, asdict
    from pathlib import Path
    from typing import Optional

    BASELINE_CANDIDATE_PATHS = [
        "C:/Users/PC/Desktop/shorts_naberal/output/zodiac-killer/final.mp4",
        "C:/Users/PC/Desktop/shorts_naberal/output/mary-celeste/final.mp4",
        "C:/Users/PC/Desktop/shorts_naberal/output/db-cooper/final.mp4",
        "C:/Users/PC/Desktop/shorts_naberal/output/elisa-lam/final.mp4",
        "C:/Users/PC/Desktop/shorts_naberal/output/kitakyushu-matsunaga/final.mp4",
    ]

    TARGET_W = 1080
    TARGET_H = 1920
    MIN_DURATION_S = 60.0  # ROADMAP Phase 16 SC#4
    MAX_DURATION_S = 125.0
    TOLERANCE = 0.10  # ±10%

    logger = logging.getLogger(__name__)


    @dataclass
    class VideoMeta:
        path: Path
        width: int
        height: int
        duration_s: float
        codec: str
        fps: float
        bitrate_kbps: int
        file_size_mb: float
        subtitle_tracks: int
        audio_channels: int  # SC#5 — stereo 확인 (expected 2)
        audio_sample_rate: int  # SC#5 — 48000 Hz 확인


    def ffprobe(path: Path) -> VideoMeta | None:
        if shutil.which("ffprobe") is None:
            return None
        r = subprocess.run(
            ["ffprobe", "-v", "error", "-show_format", "-show_streams", "-of", "json", str(path)],
            capture_output=True, text=True, encoding="utf-8", errors="replace", timeout=30,
        )
        if r.returncode != 0:
            return None
        info = json.loads(r.stdout)
        v_streams = [s for s in info.get("streams", []) if s.get("codec_type") == "video"]
        s_streams = [s for s in info.get("streams", []) if s.get("codec_type") == "subtitle"]
        a_streams = [s for s in info.get("streams", []) if s.get("codec_type") == "audio"]
        if not v_streams:
            return None
        v = v_streams[0]
        a = a_streams[0] if a_streams else {}
        fmt = info.get("format", {})
        fps_str = v.get("r_frame_rate", "30/1")
        fps_parts = fps_str.split("/")
        fps = float(fps_parts[0]) / float(fps_parts[1] or "1")
        return VideoMeta(
            path=path,
            width=int(v.get("width", 0)),
            height=int(v.get("height", 0)),
            duration_s=float(fmt.get("duration", 0.0)),
            codec=v.get("codec_name", ""),
            fps=fps,
            bitrate_kbps=int(fmt.get("bit_rate", "0")) // 1000 if fmt.get("bit_rate") else 0,
            file_size_mb=round(int(fmt.get("size", "0")) / 1024 / 1024, 2) if fmt.get("size") else 0.0,
            subtitle_tracks=len(s_streams),
            audio_channels=int(a.get("channels", 0)),
            audio_sample_rate=int(a.get("sample_rate", 0)),
        )


    def discover_baselines() -> list[Path]:
        found = []
        for p in BASELINE_CANDIDATE_PATHS:
            if Path(p).exists():
                found.append(Path(p))
            if len(found) >= 3:
                break
        return found


    def within(value: float, target: float, tolerance: float = TOLERANCE) -> bool:
        if target == 0:
            return value == 0
        return abs(value - target) / abs(target) <= tolerance


    def compare_to_baselines(ours: VideoMeta, baselines: list[VideoMeta]) -> dict:
        """각 criterion 에 대해 pass/fail + 상세 metrics."""
        report = {
            "ours": asdict(ours),
            "baselines": [asdict(b) for b in baselines],
            "criteria": {},
        }
        report["criteria"]["resolution"] = {
            "value": f"{ours.width}x{ours.height}",
            "target": f"{TARGET_W}x{TARGET_H}",
            "pass": (ours.width == TARGET_W and ours.height == TARGET_H),
        }
        report["criteria"]["duration"] = {
            "value": ours.duration_s,
            "target_min": MIN_DURATION_S,
            "target_max": MAX_DURATION_S,
            "pass": MIN_DURATION_S <= ours.duration_s <= MAX_DURATION_S,
        }
        report["criteria"]["codec"] = {
            "value": ours.codec,
            "expected_set": ["h264", "hevc"],
            "pass": ours.codec in {"h264", "hevc"},
        }
        # SC#5 audio stereo/48kHz/min-bitrate absolute thresholds (session #32 shock 519kbps defence)
        report["criteria"]["audio_channels"] = {
            "value": ours.audio_channels,
            "expected": 2,
            "pass": ours.audio_channels == 2,
        }
        report["criteria"]["audio_sample_rate"] = {
            "value": ours.audio_sample_rate,
            "expected_min": 44100,  # 44.1kHz or 48kHz acceptable
            "pass": ours.audio_sample_rate >= 44100,
        }
        report["criteria"]["min_video_bitrate"] = {
            "value": ours.bitrate_kbps,
            "threshold_kbps": 5000,  # SC#5 absolute min — session #32 shock 519kbps 재발 방지
            "pass": ours.bitrate_kbps >= 5000,
        }
        report["criteria"]["subtitle_track_present"] = {
            "value": ours.subtitle_tracks,
            "expected_min": 1,
            "pass": ours.subtitle_tracks >= 1,
        }
        if baselines:
            avg_br = sum(b.bitrate_kbps for b in baselines) / len(baselines)
            report["criteria"]["bitrate"] = {
                "value": ours.bitrate_kbps,
                "baseline_avg": avg_br,
                "tolerance": TOLERANCE,
                "pass": within(ours.bitrate_kbps, avg_br) if avg_br > 0 else True,
            }
            avg_fps = sum(b.fps for b in baselines) / len(baselines)
            report["criteria"]["fps"] = {
                "value": ours.fps,
                "baseline_avg": round(avg_fps, 2),
                "pass": within(ours.fps, avg_fps),
            }
        report["summary"] = {
            "all_pass": all(c.get("pass") for c in report["criteria"].values()),
            "fail_count": sum(1 for c in report["criteria"].values() if not c.get("pass")),
        }
        return report


    def main(argv: list[str] | None = None) -> int:
        parser = argparse.ArgumentParser(description="Phase 16-04 — final.mp4 baseline parity verifier")
        parser.add_argument("--episode", default=None, help="비교 대상 에피소드명 (기본: 우리 final.mp4)")
        parser.add_argument("--our-mp4", type=Path, required=True, help="우리 산출 final.mp4 경로")
        parser.add_argument("--report", type=Path, default=None, help="JSON report 저장 경로")
        parser.add_argument("--dry-run", action="store_true", help="baseline 없어도 우리 것만 검증")
        args = parser.parse_args(argv)

        if not args.our_mp4.exists():
            print(f"[verify_baseline_parity] our_mp4 미존재: {args.our_mp4}", file=sys.stderr)
            return 1
        ours = ffprobe(args.our_mp4)
        if ours is None:
            print("[verify_baseline_parity] ffprobe 실행 실패", file=sys.stderr)
            return 1

        baselines = []
        for bp in discover_baselines():
            meta = ffprobe(bp)
            if meta:
                baselines.append(meta)
        if not baselines and not args.dry_run:
            print("[verify_baseline_parity] baseline 3편 확보 불가 — shorts_naberal 경로 확인", file=sys.stderr)
            return 2

        report = compare_to_baselines(ours, baselines)
        report_text = json.dumps(report, indent=2, ensure_ascii=False, default=str)
        if args.report:
            args.report.parent.mkdir(parents=True, exist_ok=True)
            args.report.write_text(report_text, encoding="utf-8")
        print(report_text)
        return 0 if report["summary"]["all_pass"] else 1


    if __name__ == "__main__":
        sys.exit(main())
    ```
