#!/usr/bin/env python3
"""verify_mandatory_reads_prose.py — AGENT-STD-02 prose quality checker.

Scans 30 AGENT.md <mandatory_reads> blocks for 4 required items:
  1. FAILURES.md literal reference
  2. channel_identity.md OR channel_bible literal reference
  3. .claude/skills/ or SKILL.md path reference
  4. 한국어 literal '매 호출마다 전수 읽기, 샘플링 금지' (대표님 session #29 지시)

Exits 0 if all 30 AGENT.md pass. Exits 1 with per-file violation report.

D-A1-03 soft enforcement / AGENT-STD-02.
"""

from __future__ import annotations
import argparse
import re
import sys
from pathlib import Path

try:
    sys.stdout.reconfigure(encoding="utf-8", errors="replace")
    sys.stderr.reconfigure(encoding="utf-8", errors="replace")
except Exception:
    pass

REPO_ROOT = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(REPO_ROOT / "scripts" / "validate"))

from verify_agent_md_schema import _collect_all_agent_mds  # noqa: E402

MANDATORY_READS_BLOCK_RE = re.compile(
    r"<mandatory_reads>(.*?)</mandatory_reads>", re.DOTALL
)

# 4 필수 항목
REQUIRED_LITERALS = {
    "failures_md": r"FAILURES\.md",
    "channel_bible": r"channel_identity\.md|channel_bible",
    "skill_path": r"\.claude/skills/|SKILL\.md",
    "sampling_forbidden": r"매 호출마다 전수 읽기, 샘플링 금지",
}


def verify_file(path: Path) -> tuple[bool, list[str]]:
    """Return (ok, missing_keys). ok=True if all 4 required items present in block."""
    text = path.read_text(encoding="utf-8")
    block_match = MANDATORY_READS_BLOCK_RE.search(text)
    if not block_match:
        return False, ["(no <mandatory_reads> block — Plan 02/03 migration 필요)"]
    block = block_match.group(1)
    missing = []
    for key, pattern in REQUIRED_LITERALS.items():
        if not re.search(pattern, block):
            missing.append(key)
    return len(missing) == 0, missing


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description="AGENT-STD-02 prose quality checker")
    parser.add_argument("--all", action="store_true", default=True,
                        help="Scan all 30 producer + inspector AGENT.md (default)")
    parser.add_argument("path", type=Path, nargs="?",
                        help="Single AGENT.md (overrides --all)")
    args = parser.parse_args(argv)

    if args.path:
        targets = [args.path]
    else:
        targets = _collect_all_agent_mds()

    violations = []
    for path in targets:
        ok, missing = verify_file(path)
        if not ok:
            rel = path.relative_to(REPO_ROOT) if REPO_ROOT in path.parents else path
            violations.append((str(rel), missing))

    if violations:
        print(f"FAIL: {len(violations)}/{len(targets)} AGENT.md missing AGENT-STD-02 prose items",
              file=sys.stderr)
        for rel, missing in violations:
            print(f"  - {rel}: missing = {missing}", file=sys.stderr)
        return 1

    print(f"OK: {len(targets)}/{len(targets)} AGENT.md pass AGENT-STD-02 prose check "
          "(FAILURES.md + channel_bible + skill path + '샘플링 금지' literal)")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

"""AGENT-STD-02 — <mandatory_reads> 블록 prose 품질 검증."""

from __future__ import annotations
import sys
from pathlib import Path

import pytest

REPO_ROOT = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(REPO_ROOT / "scripts" / "validate"))

from verify_agent_md_schema import _collect_all_agent_mds  # noqa: E402
from verify_mandatory_reads_prose import (  # noqa: E402
    verify_file,
    REQUIRED_LITERALS,
    MANDATORY_READS_BLOCK_RE,
)


def test_sampling_forbidden_literal():
    """30 AGENT.md 전수에 '매 호출마다 전수 읽기, 샘플링 금지' literal 포함."""
    targets = _collect_all_agent_mds()
    assert len(targets) == 30, f"expected 30 AGENT.md, got {len(targets)}"
    missing = []
    literal = "매 호출마다 전수 읽기, 샘플링 금지"
    for p in targets:
        text = p.read_text(encoding="utf-8")
        if literal not in text:
            missing.append(str(p.relative_to(REPO_ROOT)))
    if missing:
        pytest.fail(
            f"{len(missing)}/30 AGENT.md missing '샘플링 금지' literal: {missing[:3]}..."
        )


def test_three_mandatory_items_present():
    """<mandatory_reads> 블록 에 FAILURES.md + channel_bible + skill path 3 항목 포함."""
    targets = _collect_all_agent_mds()
    violations = []
    for p in targets:
        text = p.read_text(encoding="utf-8")
        m = MANDATORY_READS_BLOCK_RE.search(text)
        if not m:
            violations.append((str(p.relative_to(REPO_ROOT)), "no block"))
            continue
        block = m.group(1)
        missing_items = []
        if "FAILURES.md" not in block:
            missing_items.append("FAILURES.md")
        # channel_bible OR channel_identity.md 둘 중 하나
        if "channel_identity.md" not in block and "channel_bible" not in block:
            missing_items.append("channel_bible/identity")
        # .claude/skills/ OR SKILL.md
        if ".claude/skills/" not in block and "SKILL.md" not in block:
            missing_items.append("skill path")
        if missing_items:
            violations.append((str(p.relative_to(REPO_ROOT)), missing_items))
    if violations:
        pytest.fail(
            f"{len(violations)}/30 AGENT.md missing items: "
            + ", ".join(f"{r}: {m}" for r, m in violations[:5])
        )


def test_all_30_agents_prose_complete():
    """verify_file 로 30 파일 전수 ok=True."""
    targets = _collect_all_agent_mds()
    fails = []
    for p in targets:
        ok, missing = verify_file(p)
        if not ok:
            fails.append((str(p.relative_to(REPO_ROOT)), missing))
    if fails:
        pytest.fail(f"{len(fails)}/30 violate AGENT-STD-02: {fails[:3]}")


def test_required_literals_count_is_4():
    """AGENT-STD-02 spec: 4 필수 항목 (대표님 session #29 확정)."""
    assert len(REQUIRED_LITERALS) == 4, \
        f"REQUIRED_LITERALS count = {len(REQUIRED_LITERALS)}, expected 4"
    assert "sampling_forbidden" in REQUIRED_LITERALS
    assert "failures_md" in REQUIRED_LITERALS
    assert "channel_bible" in REQUIRED_LITERALS
    assert "skill_path" in REQUIRED_LITERALS
```
