    ```python
    """Phase 14 Acceptance — API Adapter Remediation gate aggregator.

    Phase 14 의 5 Success Criteria 를 자동 검증:
    - SC#1: pytest tests/phase05 tests/phase06 tests/phase07 = 0 failures (ADAPT-04)

    - SC#5: pytest tests/ 전체 exit 0 (Full regression)

    추가로 harness_audit 점수 >= 80 + drift_scan A급 0 건 을 gate 에 포함.

    phase07_acceptance.py 의 패턴 (subprocess.run + stdout parse + JSON 요약) 을 복제.
    """
    from __future__ import annotations

    import json
    import os
    import subprocess
    import sys
    from datetime import datetime
    from pathlib import Path

    REPO = Path(__file__).resolve().parents[2]


    def _run(cmd: list[str], label: str) -> dict:
        """subprocess.run + stdout capture + rc record.

        Windows cp949 대응 — PYTHONIOENCODING=utf-8 강제 (RESEARCH R5).
        """
        env = os.environ.copy()
        env["PYTHONIOENCODING"] = "utf-8"
        result = subprocess.run(
            cmd,
            capture_output=True,
            text=True,
            encoding="utf-8",
            errors="replace",
            cwd=str(REPO),
            env=env,
            timeout=1200,  # 20분 — phase05/06/07 sweep 660s + buffer.
        )
        return {
            "label": label,
            "cmd": cmd,
            "rc": result.returncode,
            "stdout_tail": result.stdout[-2000:] if result.stdout else "",
            "stderr_tail": result.stderr[-500:] if result.stderr else "",
        }


    def check_sc1_phase_sweep() -> dict:
        """SC#1 — phase05/06/07 sweep 0 failures (ADAPT-04 Success Criteria #1)."""
        return _run(
            [sys.executable, "-m", "pytest", "tests/phase05", "tests/phase06", "tests/phase07",
             "--tb=no", "-q", "--no-cov"],
            "SC#1 phase05_06_07_sweep",
        )


    def check_sc2_contract_files() -> dict:
        """SC#2 — 3 contract 테스트 파일 존재."""
        files = [
            "tests/adapters/test_veo_i2v_contract.py",
            "tests/adapters/test_elevenlabs_contract.py",
            "tests/adapters/test_shotstack_contract.py",
        ]
        missing = [f for f in files if not (REPO / f).exists()]
        return {
            "label": "SC#2 contract_files_exist",
            "files": files,
            "missing": missing,
            "rc": 0 if not missing else 1,
        }


    def check_sc3_doc_exists() -> dict:
        """SC#3 — wiki/render/adapter_contracts.md 존재 + 7 adapter 행."""
        path = REPO / "wiki" / "render" / "adapter_contracts.md"
        if not path.exists():
            return {"label": "SC#3 doc_exists", "rc": 1, "reason": "file missing"}
        text = path.read_text(encoding="utf-8")
        required = ["kling_i2v", "runway_i2v", "veo_i2v", "typecast",
                    "elevenlabs", "shotstack", "whisperx"]
        missing = [name for name in required if name not in text]
        return {
            "label": "SC#3 doc_exists",
            "rc": 0 if not missing else 1,
            "missing_adapters": missing,
        }


    def check_sc4_adapter_contract_marker() -> dict:
        """SC#4 — pytest -m adapter_contract 단독 실행 green (ADAPT-06 SC#4)."""
        return _run(
            [sys.executable, "-m", "pytest", "-m", "adapter_contract", "--tb=no", "-q", "--no-cov"],

        )

        )


    def check_drift_scan() -> dict:
        """drift_scan A급 drift 0 건 (AUDIT-04 Phase 차단 조건)."""
        return _run(
            [sys.executable, "-m", "scripts.validate.drift_scan"],
            "AUDIT drift_a_class",
        )


    def main() -> int:
        checks = [
            check_sc1_phase_sweep(),
            check_sc2_contract_files(),
            check_sc3_doc_exists(),
            check_sc4_adapter_contract_marker(),
            check_sc5_full_regression(),
            check_harness_audit(),
            check_drift_scan(),
        ]
        all_ok = all(c.get("rc", 1) == 0 for c in checks)
        summary = {
            "phase": "14-api-adapter-remediation",
            "timestamp": datetime.utcnow().isoformat(timespec="seconds") + "Z",
            "all_passed": all_ok,
            "checks": checks,
        }
        out_path = REPO / ".planning" / "phases" / "14-api-adapter-remediation" / "14-05-acceptance.json"
        out_path.parent.mkdir(parents=True, exist_ok=True)
        out_path.write_text(
            json.dumps(summary, ensure_ascii=False, indent=2),
            encoding="utf-8",
        )
        print(f"Phase 14 Acceptance: {'ALL_PASS' if all_ok else 'FAIL'}")
        for c in checks:
            print(f"  {c['label']}: rc={c.get('rc', 'n/a')}")
        return 0 if all_ok else 1


    if __name__ == "__main__":
        sys.exit(main())
    ```
