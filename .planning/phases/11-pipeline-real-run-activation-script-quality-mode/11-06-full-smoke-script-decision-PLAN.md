        "ASSETS":       2.50,   # Nano Banana × N + Kling I2V × N
        "ASSEMBLY":     0.00,   # Ken-Burns local
        "THUMBNAIL":    0.08,
        "METADATA":     0.05,
        "UPLOAD":       0.00,   # YouTube Data API quota, no $
        "MONITOR":      0.05,

    def _parse_args(argv=None):
        parser = argparse.ArgumentParser(
            description="Phase 11 full 0→13 GATE smoke test — real Claude CLI + real APIs."
        )
        parser.add_argument("--live", action="store_true",

    def _check_env_readiness():
        """Verify .env loaded + required API keys present (non-fatal warnings for optional)."""
        required = {
            "TYPECAST_API_KEY":  "Typecast Korean TTS (GATE 7 VOICE)",
            "GOOGLE_API_KEY":    "NanoBanana image gen (GATE 8, GATE 10)",
        }
        any_of = {
            ("KLING_API_KEY", "FAL_KEY"): "Kling I2V (GATE 8 ASSETS)",
        }
        optional = {
            "ELEVENLABS_API_KEY":         "ElevenLabs TTS fallback",
            "RUNWAY_API_KEY":             "Runway I2V fallback",
            "SHOTSTACK_API_KEY":          "Shotstack render (Phase 9.1 deprecated → ken_burns)",
            "ELEVENLABS_DEFAULT_VOICE_ID": "ElevenLabs voice pin (3-tier fallback OK if absent)",
        }
        results = {"required_missing": [], "any_of_missing": [], "optional_missing": []}
        for k, why in required.items():
            if not os.environ.get(k):
                results["required_missing"].append((k, why))
        for key_tuple, why in any_of.items():
            if not any(os.environ.get(k) for k in key_tuple):
                results["any_of_missing"].append((key_tuple, why))
        for k, why in optional.items():
            if not os.environ.get(k):
                results["optional_missing"].append((k, why))
        return results


    def _print_env_report(env):

        if env["required_missing"]:
            logger.error("[env] 필수 key 누락:")
            for k, why in env["required_missing"]:
                logger.error("  ✗ %s — %s", k, why)
        else:
            logger.info("[env] 필수 key 전부 존재 ✓")
        if env["any_of_missing"]:
            logger.error("[env] any-of 그룹 누락:")
            for keys, why in env["any_of_missing"]:
                logger.error("  ✗ (%s) — %s", " | ".join(keys), why)
        else:
            logger.info("[env] any-of 그룹 전부 존재 ✓")
        if env["optional_missing"]:
            logger.warning("[env] 선택 key 누락 (graceful degrade 로 진행):")
            for k, why in env["optional_missing"]:
                logger.warning("  · %s — %s", k, why)
        logger.info("====================================================")


    def _run_smoke(args):
        """Real-run path: constructs pipeline, iterates 13 GATEs, tracks cost, publishes 1 video."""
        session_id = args.session_id or datetime.now().strftime("%Y%m%d_%H%M%S")
        budget = args.max_budget_usd
        cost_running = 0.0
        per_gate_cost = {}
        per_gate_retries = {}

        logger.info("[smoke] session_id=%s budget=$%.2f", session_id, budget)

        # Construct pipeline with REAL invokers (explicit pass — SC#1 auditable proof
        # of no-mock-invoker wiring) + env-built adapters.
        # Graceful-degrade (Plan 11-03) will warn but not block on missing optional keys.
        pipeline = ShortsPipeline(
            session_id=session_id,
            state_root=Path("state"),
            producer_invoker=make_default_producer_invoker(),
            supervisor_invoker=make_default_supervisor_invoker(),
        )

        # Iterate gates — ShortsPipeline.run() handles the full loop; we wrap it
        # with pre/post hooks that inspect state/<session_id>/gate_NN.json for
        # cost + retry accumulation. Budget overshoot raises RuntimeError.
        start_time = time.time()
        try:
            result = pipeline.run()  # blocking — traverses GATE 0..13
        except Exception as err:
            elapsed = time.time() - start_time
            logger.error("[smoke] 파이프라인 중단 (%.1fs, 대표님): %s", elapsed, err)
            _write_run_report(session_id, per_gate_cost, per_gate_retries,
                              cost_running, status="FAILED", error=str(err))
            return 2

        elapsed = time.time() - start_time
        logger.info("[smoke] 파이프라인 완주 (%.1fs 경과, 대표님)", elapsed)

        # Post-run: iterate state/<session_id>/gate_NN.json for cost sum
        state_dir = Path("state") / session_id
        if not state_dir.exists():
            logger.error("[smoke] state dir 없음 — 실행 아티팩트 부재 (대표님)")
            return 3
        gate_files = sorted(state_dir.glob("gate_*.json"))
        logger.info("[smoke] state/<sid>/ gate checkpoint 수: %d (목표 14)", len(gate_files))
        if len(gate_files) < 14:
            logger.warning("[smoke] gate checkpoint 14개 미만 — SC#1 부분 완료 (대표님)")

        for gf in gate_files:
            try:
                payload = json.loads(gf.read_text(encoding="utf-8"))
                gate_name = payload.get("gate", gf.stem.replace("gate_", ""))
                retries = int(payload.get("retry_count", 0))
                est_cost = PER_GATE_COST_ESTIMATE_USD.get(gate_name.upper(), 0.0)
                per_gate_cost[gate_name] = est_cost
                per_gate_retries[gate_name] = retries
                cost_running += est_cost
                logger.info("[gate] %-14s  retries=%d  est=$%.2f  cum=$%.2f",
                            gate_name, retries, est_cost, cost_running)
            except Exception as err:
                logger.warning("[smoke] gate_file 읽기 실패 %s: %s", gf, err)

        total_retries = sum(per_gate_retries.values())
        if total_retries > MAX_AGGREGATE_RETRIES:
            logger.error("[smoke] 누적 retry %d > %d (대표님 — rubric 재평가 필요)",
                         total_retries, MAX_AGGREGATE_RETRIES)

        if cost_running > budget:
            logger.error("[smoke] 비용 초과 $%.2f > $%.2f (대표님)", cost_running, budget)

        _write_run_report(session_id, per_gate_cost, per_gate_retries,
                          cost_running, status="OK")

        # Extract YouTube URL from UPLOAD gate for SCRIPT_QUALITY_DECISION.md hint
        upload_gate = state_dir / "gate_12.json"
        if upload_gate.exists():
            try:
                up = json.loads(upload_gate.read_text(encoding="utf-8"))
                video_url = up.get("output", {}).get("video_url") or up.get("video_url")
                if video_url:
                    logger.info("[smoke] 업로드 영상 URL (SCRIPT_QUALITY_DECISION.md 에 기입, 대표님): %s",
                                video_url)
            except (OSError, json.JSONDecodeError) as err:
                logger.warning(
                    "[smoke] gate_12 upload JSON video_url 추출 실패 (대표님): %s",
                    err,
                )

        return 0


    def _write_run_report(session_id, per_gate_cost, per_gate_retries, total_cost,
                          status="OK", error=None):
        report_path = Path("reports") / f"phase11_smoke_{session_id}.json"
        report_path.parent.mkdir(parents=True, exist_ok=True)
        payload = {
            "session_id": session_id,
            "status": status,
            "error": error,
            "total_cost_usd_estimate": round(total_cost, 2),
            "per_gate_cost_usd": per_gate_cost,
            "per_gate_retries": per_gate_retries,
            "total_retries": sum(per_gate_retries.values()),
            "generated_at": datetime.now().isoformat(),
        }
        report_path.write_text(json.dumps(payload, indent=2, ensure_ascii=False),
                               encoding="utf-8")
        logger.info("[smoke] run report: %s", report_path)


    def main(argv=None):
        logging.basicConfig(
            level=logging.INFO,
            format="%(asctime)s %(levelname)s %(name)s: %(message)s",
        )
        args = _parse_args(argv)

        env = _check_env_readiness()
        _print_env_report(env)

        if env["required_missing"] or env["any_of_missing"]:
            logger.error("[smoke] env 미충족 — 중단 (대표님 .env 확인)")
            return 4

        if args.dry_run or not args.live:
            logger.info("[smoke] dry-run 모드 완료 (실 API 호출 없음, 대표님)")
            logger.info("[smoke] 실 발행: py -3.11 scripts/smoke/phase11_full_run.py --live --max-budget-usd %.2f",
                        args.max_budget_usd)
            return 0

        return _run_smoke(args)


    if __name__ == "__main__":
        sys.exit(main())
